const express = require('express');
const app = express();
app.use(express.json());

// डमी डेटाबेस (असल प्रोजेक्ट में यहाँ MySQL/MongoDB जुड़ेगा)
let users = [
    { id: 1, name: "Amit", wallet: 0, views: 0, hasPack: false }
];

let appSettings = {
    minViewsForReward: 100,
    rewardAmount: 3.00
};

// 1. USER API: जब यूजर 1 वीडियो पूरा देखेगा
app.post('/api/watch-video', (req, res) => {
    const { userId } = req.body;
    let user = users.find(u => u.id === userId);

    if (!user) return res.status(404).json({ error: "User not found" });

    // अगर 100 व्यूज हो चुके हैं और पैक नहीं लिया है, तो आगे व्यूज काउंट नहीं होंगे
    if (user.views >= appSettings.minViewsForReward && !user.hasPack) {
        return res.status(403).json({ error: "Limit reached! Please buy a pack to continue earning." });
    }

    // व्यू काउंट बढ़ाएं
    user.views += 1;

    // हर 100 व्यूज पर ₹3 रिवॉर्ड जोड़ें
    if (user.views % appSettings.minViewsForReward === 0) {
        user.wallet += appSettings.rewardAmount;
    }

    res.json({ success: true, currentViews: user.views, walletBalance: user.wallet });
});

// 2. ADMIN API: एडमिन यहाँ से सेटिंग्स बदल सकता है (Admin Rights)
app.post('/api/admin/update-settings', (req, res) => {
    const { newMinViews, newRewardAmount } = req.body;
    
    if (newMinViews) appSettings.minViewsForReward = newMinViews;
    if (newRewardAmount) appSettings.rewardAmount = newRewardAmount;

    res.json({ message: "Settings updated by Admin", currentSettings: appSettings });
});

// 3. PACK BUY API: जब यूजर/डेवलपर 100 व्यूज के बाद पैक खरीदेगा
app.post('/api/buy-pack', (req, res) => {
    const { userId } = req.body;
    let user = users.find(u => u.id === userId);

    if (!user) return res.status(404).json({ error: "User not found" });

    user.hasPack = true; // पैक एक्टिवेट हो गया
    res.json({ message: "Pack purchased successfully!", hasPack: user.hasPack });
});

app.listen(3000, () => console.log('Server running on port 3000'));
