import React, { useState } from "react";
import { View, Text, TextInput, StyleSheet, SafeAreaView } from "react-native";

export default function App() {
  const [sar, setSar] = useState("1");

  // Example static rates (these will be replaced by your API call)
  const rates = {
    PKR: 74.52,
    INR: 22.3,
    NPR: 29.98,
    BDT: 31.12,
  };

  const calculate = (rate) => {
    const value = parseFloat(sar);
    if (isNaN(value)) return "0";
    return (value * rate).toFixed(2);
  };

  return (
    <SafeAreaView style={styles.container}>
      <Text style={styles.title}>Saudi Riyal Converter</Text>

      <TextInput
        style={styles.input}
        value={sar}
        onChangeText={setSar}
        keyboardType="numeric"
        placeholder="Enter SAR"
      />

      <View style={styles.card}>
        <Text style={styles.label}>Pakistan (PKR):</Text>
        <Text style={styles.value}>{calculate(rates.PKR)}</Text>
      </View>

      <View style={styles.card}>
        <Text style={styles.label}>India (INR):</Text>
        <Text style={styles.value}>{calculate(rates.INR)}</Text>
      </View>

      <View style={styles.card}>
        <Text style={styles.label}>Nepal (NPR):</Text>
        <Text style={styles.value}>{calculate(rates.NPR)}</Text>
      </View>

      <View style={styles.card}>
        <Text style={styles.label}>Bangladesh (BDT):</Text>
        <Text style={styles.value}>{calculate(rates.BDT)}</Text>
      </View>
    </SafeAreaView>
  );
}
