import React, { useState } from 'react';
import { View, Text, FlatList, TouchableOpacity, StyleSheet, Button } from 'react-native';
import MapView, { Marker } from 'react-native-maps';

// Данные парковок
const parkingSpots = [
  { id: '1', name: 'ТЦ Алмалы', address: 'ул. Абая 10', available: 5, price: 300, latitude: 43.2389, longitude: 76.8897 },
  { id: '2', name: 'Автовокзал', address: 'ул. Жамбыла', available: 2, price: 200, latitude: 43.2356, longitude: 76.8895 },
  { id: '3', name: 'Университет', address: 'ул. Толе Би', available: 10, price: 250, latitude: 43.2423, longitude: 76.8972 },
];

export default function App() {
  const [selectedSpot, setSelectedSpot] = useState(null);

  if (selectedSpot) {
    return (
      <View style={styles.container}>
        <Text style={styles.title}>{selectedSpot.name}</Text>
        <Text>Адрес: {selectedSpot.address}</Text>
        <Text>Свободных мест: {selectedSpot.available}</Text>
        <Text>Цена (1 час): {selectedSpot.price} KZT</Text>
        <Button title="Назад" onPress={() => setSelectedSpot(null)} color="#4F46E5" />
      </View>
    );
  }

  return (
    <View style={{ flex: 1 }}>
      <MapView
        style={{ flex: 1 }}
        initialRegion={{
          latitude: 43.2389,
          longitude: 76.8897,
          latitudeDelta: 0.02,
          longitudeDelta: 0.02,
        }}
      >
        {parkingSpots.map(spot => (
          <Marker
            key={spot.id}
            coordinate={{ latitude: spot.latitude, longitude: spot.longitude }}
            title={spot.name}
            description={`${spot.available} свободных мест`}
          />
        ))}
      </MapView>

      <FlatList
        data={parkingSpots}
        keyExtractor={item => item.id}
        style={styles.list}
        renderItem={({ item }) => (
          <TouchableOpacity style={styles.card} onPress={() => setSelectedSpot(item)}>
            <Text style={styles.title}>{item.name}</Text>
            <Text>{item.address}</Text>
            <Text>Свободно: {item.available}</Text>
            <Text>Цена: {item.price} KZT/час</Text>
          </TouchableOpacity>
        )}
      />
    </View>
  );
}

const styles = StyleSheet.create({
  list: {
    backgroundColor: '#fff',
  },
  card: {
    padding: 10,
    borderBottomWidth: 1,
    borderColor: '#ddd',
  },
  container: {
    flex: 1,
    justifyContent: 'center',
    alignItems: 'center',
    padding: 16,
  },
  title: {
    fontSize: 18,
    fontWeight: 'bold',
  },
});
