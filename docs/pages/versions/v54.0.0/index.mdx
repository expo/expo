import React, { useState, useEffect } from "react";
import {
  View,
  Text,
  TextInput,
  Pressable,
  FlatList,
  StyleSheet,
} from "react-native";
import AsyncStorage from "@react-native-async-storage/async-storage";

/* ================= APP ================= */
export default function App() {
  const [screen, setScreen] = useState("login");
  const [user, setUser] = useState("");
  const [balance, setBalance] = useState(125000);
  const [amount, setAmount] = useState("");
  const [receiver, setReceiver] = useState("");
  const [transactions, setTransactions] = useState([]);

  /* ===== Load Data ===== */
  useEffect(() => {
    loadData();
  }, []);

  const loadData = async () => {
    const savedBalance = await AsyncStorage.getItem("balance");
    const savedTx = await AsyncStorage.getItem("tx");

    if (savedBalance) setBalance(Number(savedBalance));
    if (savedTx) setTransactions(JSON.parse(savedTx));
  };

  const saveData = async (newBalance, newTx) => {
    await AsyncStorage.setItem("balance", String(newBalance));
    await AsyncStorage.setItem("tx", JSON.stringify(newTx));
  };

  /* ===== LOGIN ===== */
  if (screen === "login") {
    return (
      <View style={styles.container}>
        <Text style={styles.title}>💳 Y Pay Bank</Text>

        <TextInput
          placeholder="اسم المستخدم"
          value={user}
          onChangeText={setUser}
          style={styles.input}
        />

        <Pressable
          style={styles.button}
          onPress={() => setScreen("home")}
        >
          <Text style={styles.buttonText}>دخول</Text>
        </Pressable>
      </View>
    );
  }

  /* ===== HOME ===== */
  if (screen === "home") {
    return (
      <View style={styles.container}>
        <Text style={styles.title}>مرحباً {user} 👋</Text>

        <View style={styles.card}>
          <Text>الرصيد</Text>
          <Text style={styles.balance}>
            {balance.toLocaleString("ar-YE")} ر.ي
          </Text>
        </View>

        <View style={styles.row}>
          <Pressable
            style={styles.smallBtn}
            onPress={() => setScreen("transfer")}
          >
            <Text>💸 تحويل</Text>
          </Pressable>

          <Pressable
            style={styles.smallBtn}
            onPress={() => setScreen("profile")}
          >
            <Text>👤 حسابي</Text>
          </Pressable>
        </View>

        <Text style={styles.subtitle}>آخر العمليات</Text>

        <FlatList
          data={transactions}
          keyExtractor={(item, i) => i.toString()}
          renderItem={({ item }) => (
            <View style={styles.tx}>
              <Text>{item.text}</Text>
              <Text style={{ color: item.amount > 0 ? "green" : "red" }}>
                {item.amount}
              </Text>
            </View>
          )}
        />
      </View>
    );
  }

  /* ===== TRANSFER ===== */
  if (screen === "transfer") {
    const sendMoney = async () => {
      if (!receiver || !amount) return;

      const value = Number(amount);
      if (value > balance) {
        alert("رصيد غير كافي");
        return;
      }

      const newBalance = balance - value;

      const newTx = [
        {
          text: `تحويل إلى ${receiver}`,
          amount: -value,
        },
        ...transactions,
      ];

      setBalance(newBalance);
      setTransactions(newTx);

      await saveData(newBalance, newTx);

      setReceiver("");
      setAmount("");
      setScreen("home");
    };

    return (
      <View style={styles.container}>
        <Text style={styles.title}>💸 تحويل</Text>

        <TextInput
          placeholder="المستلم"
          value={receiver}
          onChangeText={setReceiver}
          style={styles.input}
        />

        <TextInput
          placeholder="المبلغ"
          value={amount}
          onChangeText={setAmount}
          keyboardType="numeric"
          style={styles.input}
        />

        <Pressable style={styles.button} onPress={sendMoney}>
          <Text style={styles.buttonText}>إرسال</Text>
        </Pressable>

        <Pressable onPress={() => setScreen("home")}>
          <Text style={{ color: "blue", marginTop: 10 }}>رجوع</Text>
        </Pressable>
      </View>
    );
  }

  /* ===== PROFILE ===== */
  return (
    <View style={styles.container}>
      <Text style={styles.title}>👤 الحساب</Text>

      <Text style={styles.card}>اسم المستخدم: {user}</Text>

      <Pressable
        style={[styles.button, { backgroundColor: "red" }]}
        onPress={() => setScreen("login")}
      >
        <Text style={styles.buttonText}>تسجيل خروج</Text>
      </Pressable>
    </View>
  );
}
