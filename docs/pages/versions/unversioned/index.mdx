type Role = "owner" | "engineer" | "supervisor" | "contractor" | "admin";

type SessionState = {
  token?: string;
  role?: Role;
  setSession: (t?: string, r?: Role) => void;
  clear: () => void;
};

export const useSession = create<SessionState>((set) => ({
  token: undefined,
  role: undefined,
  setSession: (t, r) => set({ token: t, role: r }),
  clear: () => set({ token: undefined, role: undefined })
}));

4.4 lib/forms/quote.ts

import { z } from "zod";

export const quoteItemSchema = z.object({
  description: z.string().min(2),
  qty: z.number().positive(),
  unit: z.string().optional(),
  rate: z.number().nonnegative()
});

export const quoteSchema = z.object({
  job_id: z.string().uuid(),
  contractor_id: z.string().uuid(),
  items: z.array(quoteItemSchema).min(1),
  discount: z.number().nonnegative().default(0),
  vat: z.number().nonnegative().default(0)
}).transform(v => {
  const subtotal = v.items.reduce((s,i)=> s + i.qty * i.rate, 0);
  const total = subtotal - v.discount + v.vat;
  return { ...v, subtotal, total };
});

4.6 app/(tabs)/jobs.tsx

import { View, Text, FlatList } from "react-native";
import { useQuery } from "@tanstack/react-query";
import { api } from "@/lib/api";

export default function Jobs() {
  const { data, isLoading, error } = useQuery({
    queryKey: ["jobs"],
    queryFn: async () => (await api.get("/jobs")).data
  });

  if (isLoading) return <Text>Loading‚Ä¶</Text>;
  if (error) return <Text>Failed to load.</Text>;

  return (
    <FlatList
      data={data}
      keyExtractor={(item) => item.id}
      renderItem={({ item }) => (
        <View style={{ padding: 12 }}>
          <Text>{item.title}</Text>
          <Text>{item.status}</Text>
        </View>
      )}
    />
  );
}
