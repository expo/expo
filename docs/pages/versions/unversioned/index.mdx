// --- 3. CREATE POST COMPONENT ---
const CreatePost = ({ onPostSubmit }) => {
  const [content, setContent] = useState('');

  const handleSubmit = (e) => {
    e.preventDefault();
    if (content.trim()) {
      onPostSubmit(content.trim());
      setContent(''); // Clear the input after submission
    }
  };

  return (
    <div style={styles.createPostCard}>
      <form onSubmit={handleSubmit}>
        <textarea
          style={styles.postInput}
          placeholder="What's on your mind, You?"
          value={content}
          onChange={(e) => setContent(e.target.value)}
        />
        <button 
          type="submit" 
          style={styles.postSubmitButton} 
          disabled={!content.trim()}
        >
          Post
        </button>
      </form>
    </div>
  );
};

// --- 4. POST COMPONENT (Unchanged) ---
const Post = ({ post, onToggleLike, onAddComment }) => {
  const [newComment, setNewComment] = useState('');
  
  const handleCommentSubmit = (e) => {
    e.preventDefault();
    if (newComment.trim()) {
      onAddComment(post.id, newComment.trim());
      setNewComment('');
    }
  };

  const likeButtonStyle = post.isLiked 
    ? { ...styles.actionButton, ...styles.likedButton }
    : styles.actionButton;

  return (
    <div style={styles.postCard}>
      {/* Header */}
      <div style={styles.postHeader}>
        <span style={styles.postUser}>{post.user}</span>
        <span style={styles.postTimestamp}>{post.timestamp}</span>
      </div>

      {/* Content */}
      <p style={styles.postContent}>{post.content}</p>

      {/* Likes Count */}
      <div style={{ color: '#606770', fontSize: '12px', marginBottom: '10px' }}>
        {post.likes} {post.likes === 1 ? 'Like' : 'Likes'}
      </div>

      {/* Action Bar (Like/Comment) */}
      <div style={styles.actionBar}>
        <button 
          style={likeButtonStyle} 
          onClick={() => onToggleLike(post.id)}
        >
          {post.isLiked ? 'üëç Liked' : 'üëç Like'}
        </button>
        <button style={styles.actionButton}>
          üí¨ Comment
        </button>
      </div>

      {/* Comments Section */}
      <div style={styles.commentsSection}>
        {post.comments.map(comment => (
          <div key={comment.id} style={styles.comment}>
            <span style={styles.commentUser}>{comment.user}</span>
            {comment.text}
          </div>
        ))}

        {/* New Comment Input */}
        <form style={styles.newCommentForm} onSubmit={handleCommentSubmit}>
          <input
            type="text"
            placeholder="Write a comment..."
            value={newComment}
            onChange={(e) => setNewComment(e.target.value)}
            style={styles.commentInput}
          />
          <button type="submit" style={styles.commentButton}>Post</button>
        </form>
      </div>
    </div>
  );
};

// --- 5. MAIN APP COMPONENT ---
const SocialFeedApp = () => {
  const [posts, setPosts] = useState(initialPosts);

  // Function to add a brand new post
  const addNewPost = (content) => {
    const newPost = {
      id: Date.now(), // Unique ID
      user: "Current User", // Mock the current user
      timestamp: "Just now", 
      content: content,
      likes: 0,
      isLiked: false,
      comments: [],
    };
    // Add new post to the beginning of the array
    setPosts(prevPosts => [newPost, ...prevPosts]); 
  };

  // Function to handle liking/unliking a post
  const toggleLike = (postId) => {
    setPosts(prevPosts =>
      prevPosts.map(post => {
        if (post.id === postId) {
          const newIsLiked = !post.isLiked;
          const newLikes = newIsLiked ? post.likes + 1 : post.likes - 1;
          return { ...post, isLiked: newIsLiked, likes: newLikes };
        }
        return post;
      })
    );
  };

  // Function to handle adding a new comment
  const addComment = (postId, commentText) => {
    setPosts(prevPosts =>
      prevPosts.map(post => {
        if (post.id === postId) {
          const newComment = {
            id: Date.now(), 
            user: "You", 
            text: commentText,
          };
          return { 
            ...post, 
            comments: [...post.comments, newComment] 
          };
        }
        return post;
      })
    );
  };

  return (
    <div style={styles.appContainer}>
      <div style={styles.header}>
        My Social Feed
      </div>
      
      {/* NEW: Post Creation Form */}
      <CreatePost onPostSubmit={addNewPost} />

      {/* The Feed */}
      {posts.map(post => (
        <Post 
          key={post.id} 
          post={post} 
          onToggleLike={toggleLike} 
          onAddComment={addComment} 
        />
      ))}
    </div>
  );
};

export default SocialFeedApp;
