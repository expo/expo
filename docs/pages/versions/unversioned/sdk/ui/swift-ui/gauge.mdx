<Tabs>
  <Tab label="Preview">
    <ContentSpotlight

  <Tab label="Code">
    ```tsx
    import { Gauge, Host } from "@expo/ui/swift-ui";

    <Host matchContents>
      <Gauge
        max={{ value: 1, label: '1' }}
        min={{ value: 0, label: '0' }}
        current={{ value: 0.5 }}
        color={[
          PlatformColor('systemRed'),
          PlatformColor('systemOrange'),
          PlatformColor('systemYellow'),
          PlatformColor('systemGreen'),
        ]}
        type="circularCapacity"
      />
    </Host>
    ```
