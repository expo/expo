## Usage

```tsx
import { Divider, Host, VStack, Text } from '@expo/ui/swift-ui';

function DividerExample() {
  return (
    <Host style={{ flex: 1 }}>
      <VStack>
        <Text>First section</Text>
        <Divider />
        <Text>Second section</Text>
        <Divider />
        <Text>Third section</Text>
      </VStack>
    </Host>
  );
}
```

> See [official SwiftUI documentation](https://developer.apple.com/documentation/swiftui/divider) for more information.
