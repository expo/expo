Expo UI Chart wraps Apple's [Swift Charts](https://developer.apple.com/documentation/charts) framework, and supports rendering line, point, bar, area, pie, and rectangle charts with reference lines.

> **info** `Chart` requires iOS 16+ and tvOS 16+. Pie charts (`type="pie"`) additionally require iOS 17+ and tvOS 17+.

## Usage

### Line chart

```tsx LineChartExample.tsx
import { Host, Chart } from '@expo/ui/swift-ui';

export default function LineChartExample() {
  return (
    <Host style={{ flex: 1 }}>
      <Chart
        type="line"
        data={[
          { x: 'Jan', y: 15 },
          { x: 'Feb', y: 25 },
          { x: 'Mar', y: 18 },
          { x: 'Apr', y: 32 },
        ]}
        style={{ flex: 1 }}
      />
    </Host>
  );
}
```

### Bar chart

```tsx BarChartExample.tsx
import { Host, Chart } from '@expo/ui/swift-ui';

export default function BarChartExample() {
  return (
    <Host style={{ flex: 1 }}>
      <Chart
        type="bar"
        data={[
          { x: 'Q1', y: 75 },
          { x: 'Q2', y: 85 },
          { x: 'Q3', y: 65 },
          { x: 'Q4', y: 95 },
        ]}
        showLegend
        style={{ flex: 1 }}
      />
    </Host>
  );
}
```

### Pie chart

```tsx PieChartExample.tsx
import { Host, Chart } from '@expo/ui/swift-ui';

export default function PieChartExample() {
  return (
    <Host style={{ flex: 1 }}>
      <Chart
        type="pie"
        data={[
          { x: 'Direct', y: 45, color: '#6366F1' },
          { x: 'Social', y: 30, color: '#EC4899' },
          { x: 'Email', y: 25, color: '#10B981' },
        ]}
        pieStyle={{ innerRadius: 0.5, angularInset: 2 }}
        showLegend
        style={{ flex: 1 }}
      />
    </Host>
  );
}
```

### Point and area charts

```tsx PointAreaExample.tsx
import { Host, Chart, VStack } from '@expo/ui/swift-ui';

const data = [
  { x: 0, y: 10 },
  { x: 1.5, y: 25 },
  { x: 3, y: 18 },
  { x: 4.5, y: 32 },
];

export default function PointAreaExample() {
  return (
    <Host style={{ flex: 1 }}>
      <VStack spacing={16}>
        <Chart type="point" data={data} pointStyle={{ pointStyle: 'diamond', pointSize: 8 }} style={{ height: 150 }} />
        <Chart type="area" data={data} areaStyle={{ color: '#10B981' }} style={{ height: 150 }} />
      </VStack>
    </Host>
  );
}
```
