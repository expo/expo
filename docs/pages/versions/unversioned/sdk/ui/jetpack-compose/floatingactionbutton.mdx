Expo UI `FloatingActionButton` wraps Material Design 3's [`FloatingActionButton`](https://developer.android.com/develop/ui/compose/components/fab) and [`ExtendedFloatingActionButton`](https://developer.android.com/develop/ui/compose/components/fab#extended). When `label` is absent a standard FAB is rendered; when `label` is present an `ExtendedFloatingActionButton` with animated label expansion is rendered.

## Usage

### Standard FAB

```tsx StandardFABExample.tsx
import { FloatingActionButton, Host } from '@expo/ui/jetpack-compose';

export default function StandardFABExample() {
  return (
    <Host matchContents>
      <FloatingActionButton
        icon={require('./assets/add.xml')}
        onPress={() => console.log('FAB pressed')}
      />
    </Host>
  );
}
```

### FAB sizes

```tsx FABSizesExample.tsx
import { FloatingActionButton, Host } from '@expo/ui/jetpack-compose';
import { View } from 'react-native';

export default function FABSizesExample() {
  return (
    <View style={{ flexDirection: 'row', gap: 16 }}>
      <Host matchContents>
        <FloatingActionButton icon={require('./assets/add.xml')} size="small" onPress={() => {}} />
      </Host>
      <Host matchContents>
        <FloatingActionButton icon={require('./assets/add.xml')} size="medium" onPress={() => {}} />
      </Host>
      <Host matchContents>
        <FloatingActionButton icon={require('./assets/add.xml')} size="large" onPress={() => {}} />
      </Host>
    </View>
  );
}
```

### Extended FAB

```tsx ExtendedFABExample.tsx
import { FloatingActionButton, Host } from '@expo/ui/jetpack-compose';
import { useState } from 'react';

export default function ExtendedFABExample() {
  const [expanded, setExpanded] = useState(true);

  return (
    <Host matchContents>
      <FloatingActionButton
        icon={require('./assets/edit.xml')}
        label="Edit"
        expanded={expanded}
        onPress={() => setExpanded((v) => !v)}
      />
    </Host>
  );
}
```

### Custom color

```tsx FABCustomColorExample.tsx
import { FloatingActionButton, Host } from '@expo/ui/jetpack-compose';

export default function FABCustomColorExample() {
  return (
    <Host matchContents>
      <FloatingActionButton
        icon={require('./assets/add.xml')}
        label="New item"
        containerColor="#E8DEF8"
        onPress={() => console.log('pressed')}
      />
    </Host>
  );
}
```
