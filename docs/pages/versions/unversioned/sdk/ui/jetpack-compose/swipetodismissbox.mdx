Expo UI SwipeToDismissBox matches the official Jetpack Compose [`SwipeToDismissBox`](https://developer.android.com/develop/ui/compose/touch-input/user-interactions/swipe-to-dismiss) API. It wraps any content and provides swipe-to-dismiss gestures with customizable background content.

## Usage

### Swipe to delete

Swipe right-to-left to delete an item. The background grows with the swipe gesture.

```tsx SwipeToDelete.tsx
import { useState } from 'react';
import {
  Host,
  LazyColumn,
  ListItem,
  Box,
  Icon,
  SwipeToDismissBox,
  Text,
} from '@expo/ui/jetpack-compose';
import {
  fillMaxSize,
  fillMaxWidth,
  clip,
  background,
  paddingAll,
  Shapes,
} from '@expo/ui/jetpack-compose/modifiers';

// Replace with your own vector drawable asset
const deleteIcon = require('./assets/delete.xml');

export default function SwipeToDelete() {
  const [items, setItems] = useState([
    { id: '1', title: 'First item' },
    { id: '2', title: 'Second item' },
    { id: '3', title: 'Third item' },
  ]);

  return (
    <Host style={{ flex: 1 }}>
      <LazyColumn modifiers={[fillMaxWidth()]}>
        {items.map(item => (
          <SwipeToDismissBox
            key={item.id}
            enableDismissFromStartToEnd={false}
            onEndToStart={() => setItems(prev => prev.filter(i => i.id !== item.id))}>
            <SwipeToDismissBox.BackgroundContent>
              <Box
                contentAlignment="center"
                modifiers={[
                  fillMaxSize(),
                  clip(Shapes.RoundedCorner(24)),
                  background('#EF5350'),
                  paddingAll(16),
                ]}>
                <Icon source={deleteIcon} size={24} tint="#FFFFFF" />
              </Box>
            </SwipeToDismissBox.BackgroundContent>
            <ListItem modifiers={[fillMaxWidth()]}>
              <ListItem.HeadlineContent>
                <Text>{item.title}</Text>
              </ListItem.HeadlineContent>
            </ListItem>
          </SwipeToDismissBox>
        ))}
      </LazyColumn>
    </Host>
  );
}
```

### Different backgrounds per direction

Use `BackgroundStartToEnd` and `BackgroundEndToStart` to show different backgrounds depending on the swipe direction.

```tsx SwipeBothWays.tsx
import { useState } from 'react';
import {
  Host,
  LazyColumn,
  ListItem,
  Box,
  Icon,
  SwipeToDismissBox,
  Text,
} from '@expo/ui/jetpack-compose';
import {
  fillMaxSize,
  fillMaxWidth,
  clip,
  background,
  paddingAll,
  Shapes,
} from '@expo/ui/jetpack-compose/modifiers';

// Replace with your own vector drawable asset
const archiveIcon = require('./assets/archive.xml');
const deleteIcon = require('./assets/delete.xml');

export default function SwipeBothWays() {
  const [items, setItems] = useState([{ id: '1', title: 'Swipe either way' }]);

  return (
    <Host style={{ flex: 1 }}>
      <LazyColumn modifiers={[fillMaxWidth()]}>
        {items.map(item => (
          <SwipeToDismissBox
            key={item.id}
            onStartToEnd={() => setItems(prev => prev.filter(i => i.id !== item.id))}
            onEndToStart={() => setItems(prev => prev.filter(i => i.id !== item.id))}>
            <SwipeToDismissBox.BackgroundStartToEnd>
              <Box
                contentAlignment="center"
                modifiers={[
                  fillMaxSize(),
                  clip(Shapes.RoundedCorner(24)),
                  background('#4CAF50'),
                  paddingAll(16),
                ]}>
                <Icon source={archiveIcon} size={24} tint="#FFFFFF" />
              </Box>
            </SwipeToDismissBox.BackgroundStartToEnd>
            <SwipeToDismissBox.BackgroundEndToStart>
              <Box
                contentAlignment="center"
                modifiers={[
                  fillMaxSize(),
                  clip(Shapes.RoundedCorner(24)),
                  background('#EF5350'),
                  paddingAll(16),
                ]}>
                <Icon source={deleteIcon} size={24} tint="#FFFFFF" />
              </Box>
            </SwipeToDismissBox.BackgroundEndToStart>
            <ListItem modifiers={[fillMaxWidth()]}>
              <ListItem.HeadlineContent>
                <Text>{item.title}</Text>
              </ListItem.HeadlineContent>
            </ListItem>
          </SwipeToDismissBox>
        ))}
      </LazyColumn>
    </Host>
  );
}
```
