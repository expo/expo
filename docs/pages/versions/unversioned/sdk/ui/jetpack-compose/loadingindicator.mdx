    <ContentSpotlight

  <Tab label="Code">
    ```tsx
    import { LoadingIndicator } from '@expo/ui/jetpack-compose';
     {/* Indeterminate (continuous animation) */}
     <LoadingIndicator />

     {/* Determinate with progress value */}
     <LoadingIndicator progress={0.5} color="#6200EE" />

     {/* Contained variant with background */}
     <LoadingIndicator
       variant="contained"
       containerColor="#E8DEF8"
       color="#6200EE"
     />
     ```
