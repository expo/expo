import React, { useState } from 'react';
import { View, Text, TextInput, Button, FlatList, StyleSheet, Alert } from 'react-native';

export default function App() {
  const [screen, setScreen] = useState('login'); // login, cadastro, home
  const [username, setUsername] = useState('');
  const [password, setPassword] = useState('');
  const [products, setProducts] = useState([]);
  const [newProduct, setNewProduct] = useState('');
  const [denunciation, setDenunciation] = useState('');

  // Usuário simples para teste
  const testUser = { username: 'user', password: '1234' };

  const handleLogin = () => {
    if (username === testUser.username && password === testUser.password) {
      setScreen('home');
    } else {
      Alert.alert('Erro', 'Usuário ou senha incorretos');
    }
  };

  const handleAddProduct = () => {
    if (newProduct.trim() !== '') {
      setProducts([...products, { key: Math.random().toString(), name: newProduct }]);
      setNewProduct('');
    }
  };

  const handleDenounce = () => {
    if (denunciation.trim() !== '') {
      Alert.alert('Denúncia enviada', denunciation);
      setDenunciation('');
    }
  };

  if (screen === 'login') {
    return (
      <View style={styles.container}>
        <Text style={styles.title}>VCS - Login</Text>
        <TextInput
          placeholder="Usuário"
          style={styles.input}
          value={username}
          onChangeText={setUsername}
        />
        <TextInput
          placeholder="Senha"
          style={styles.input}
          secureTextEntry
          value={password}
          onChangeText={setPassword}
        />
        <Button title="Entrar" onPress={handleLogin} />
        <Button title="Cadastro" onPress={() => setScreen('cadastro')} />
      </View>
    );
  }

  if (screen === 'cadastro') {
    return (
      <View style={styles.container}>
        <Text style={styles.title}>Cadastro (simulado)</Text>
        <TextInput placeholder="Usuário" style={styles.input} />
        <TextInput placeholder="Senha" style={styles.input} secureTextEntry />
        <Button title="Cadastrar" onPress={() => { Alert.alert('Cadastro feito'); setScreen('login'); }} />
      </View>
    );
  }

  return (
    <View style={styles.container}>
      <Text style={styles.title}>VCS - Home</Text>

      <Text style={styles.subtitle}>Adicionar Produto:</Text>
      <TextInput
        placeholder="Nome do produto"
        style={styles.input}
        value={newProduct}
        onChangeText={setNewProduct}
      />
      <Button title="Adicionar" onPress={handleAddProduct} />

      <FlatList
        data={products}
        renderItem={({ item }) => <Text style={styles.item}>{item.name}</Text>}
      />

      <Text style={styles.subtitle}>Denunciar Produto:</Text>
      <TextInput
        placeholder="Descrição"
        style={styles.input}
        value={denunciation}
        onChangeText={setDenunciation}
      />
      <Button title="Enviar Denúncia" onPress={handleDenounce} />
      
      <Button title="Sair" onPress={() => setScreen('login')} color="red" />
    </View>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1, padding: 20, justifyContent: 'center' },
  title: { fontSize: 28, fontWeight: 'bold', marginBottom: 20, textAlign: 'center' },
  subtitle: { fontSize: 18, marginTop: 20, marginBottom: 5 },
  input: { borderWidth: 1, borderColor: '#ccc', padding: 10, marginBottom: 10, borderRadius: 5 },
  item: { fontSize: 16, padding: 5, borderBottomWidth: 1, borderBottomColor: '#ddd' },
});
