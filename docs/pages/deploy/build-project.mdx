      // Save reminders to localStorage
      localStorage.setItem('morningReminder', morningTime);

      // Schedule notifications

    // Schedule notifications
    function scheduleNotifications(morningTime, eveningTime) {
      if (!('Notification' in window)) {
        alert('Your browser does not support notifications.');
        return;
      }

      if (Notification.permission !== 'granted') {
        Notification.requestPermission().then((permission) => {
          if (permission === 'granted') {
            alert('Notification permission granted.');
          }
        });
      }

      // Clear existing reminders
      clearInterval(window.morningReminderInterval);
      clearInterval(window.eveningReminderInterval);

      // Morning reminder
      const morningParts = morningTime.split(':');
      const morningHours = parseInt(morningParts[0]);
      const morningMinutes = parseInt(morningParts[1]);
      window.morningReminderInterval = setInterval(() => {
        const now = new Date();
        if (now.getHours() === morningHours && now.getMinutes() === morningMinutes) {
          new Notification('Good Morning!', {
            body: "It's time to pray and read your Bible.",
          });
        }
      }, 60000); // Check every minute

      // Evening reminder
      const eveningParts = eveningTime.split(':');
      const eveningHours = parseInt(eveningParts[0]);
      const eveningMinutes = parseInt(eveningParts[1]);
      window.eveningReminderInterval = setInterval(() => {
        const now = new Date();
        if (now.getHours() === eveningHours && now.getMinutes() === eveningMinutes) {
          new Notification('Good Evening!', {
            body: "It's time to pray and reflect on God's Word.",
          });
        }
      }, 60000); // Check every minute
    }

    // Load saved reminders on page load

      const savedMorningTime = localStorage.getItem('morningReminder');
      const savedEveningTime = localStorage.getItem('eveningReminder');
      if (savedMorningTime && savedEveningTime) {
        document.getElementById('morningTime').value = savedMorningTime;
        document.getElementById('eveningTime').value = savedEveningTime;
        scheduleNotifications(savedMorningTime, savedEveningTime);
      }
    });
