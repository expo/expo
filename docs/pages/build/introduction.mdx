import React, { useState, useEffect } from "react";
import {
  SafeAreaView,
  View,
  Text,
  FlatList,
  ScrollView,
  TextInput,
  Button,
  TouchableOpacity,
  Dimensions,
  StyleSheet,
} from "react-native";
import AsyncStorage from "@react-native-async-storage/async-storage";
import { NavigationContainer, DefaultTheme, DarkTheme } from "@react-navigation/native";
import { createBottomTabNavigator } from "@react-navigation/bottom-tabs";
import { LineChart } from "react-native-chart-kit";
import Icon from "react-native-vector-icons/MaterialIcons";

// === CONFIG ===
const ALPHA_KEY = "demo"; // ganti dengan API key Alpha Vantage kamu
const NEWS_KEY = "demo";  // ganti dengan API key NewsAPI kamu

const Tab = createBottomTabNavigator();
const screenWidth = Dimensions.get("window").width;

/* =================== MARKET SCREEN =================== */
function MarketScreen() {
  const [markets, setMarkets] = useState([]);
  const [offline, setOffline] = useState(false);

  const symbols = [
    { name: "Dow Jones", code: "DJI" },
    { name: "NASDAQ", code: "IXIC" },
    { name: "S&P 500", code: "GSPC" },
    { name: "Hang Seng", code: "HSI" },
    { name: "Nikkei 225", code: "N225" },
    { name: "IHSG", code: "JKSE" },
  ];

  const fetchMarkets = async () => {
    try {
      const results = [];
      for (let s of symbols) {
        const url = `https://www.alphavantage.co/query?function=TIME_SERIES_DAILY&symbol=${s.code}&apikey=${ALPHA_KEY}`;
        const res = await fetch(url);
        const data = await res.json();
        const ts = data["Time Series (Daily)"];
        if (ts) {
          const last = Object.values(ts)[0];
          const price = parseFloat(last["4. close"]);
          results.push({ ...s, price });
        }
      }
      setMarkets(results);
      await AsyncStorage.setItem("markets", JSON.stringify(results));
    } catch (e) {
      console.log("No internet, loading cache...");
      const cache = await AsyncStorage.getItem("markets");
      if (cache) {
        setMarkets(JSON.parse(cache));
        setOffline(true);
      }
    }
  };

  useEffect(() => {
    fetchMarkets();
  }, []);

  return (
    <SafeAreaView style={styles.container}>
      <Text style={styles.header}>🌍 Global Markets</Text>
      {offline && <Text style={styles.offline}>📴 Offline Mode</Text>}
      <FlatList
        data={markets}
        keyExtractor={(item) => item.code}
        renderItem={({ item }) => (
          <View style={styles.card}>
            <Text style={styles.title}>{item.name}</Text>
            <Text style={styles.text}>{item.price || "--"}</Text>
          </View>
        )}
      />
    </SafeAreaView>
  );
}

/* =================== STOCKS SCREEN =================== */
function StocksScreen() {
  const [symbol, setSymbol] = useState("AAPL");
  const [prices, setPrices] = useState([]);
  const [offline, setOffline] = useState(false);

  const fetchStock = async () => {
    try {
      const url = `https://www.alphavantage.co/query?function=TIME_SERIES_DAILY&symbol=${symbol}&apikey=${ALPHA_KEY}`;
      const res = await fetch(url);
      const data = await res.json();
      const ts = data["Time Series (Daily)"];
      if (ts) {
        const close = Object.values(ts).map((d) => parseFloat(d["4. close"])).slice(0, 7).reverse();
        setPrices(close);
        await AsyncStorage.setItem("stock_" + symbol, JSON.stringify(close));
      }
    } catch (e) {
      const cache = await AsyncStorage.getItem("stock_" + symbol);
      if (cache) {
        setPrices(JSON.parse(cache));
        setOffline(true);
      }
    }
  };

  useEffect(() => {
    fetchStock();
  }, [symbol]);

  return (
    <ScrollView style={styles.container}>
      <Text style={styles.header}>📈 Stock Detail</Text>
      <TextInput
        style={styles.input}
        placeholder="Enter symbol (ex: AAPL)"
        value={symbol}
        onChangeText={setSymbol}
      />
      {offline && <Text style={styles.offline}>📴 Offline Mode</Text>}
      {prices.length > 0 && (
        <LineChart
          data={{
            labels: ["Day1", "2", "3", "4", "5", "6", "7"],
            datasets: [{ data: prices }],
          }}
          width={screenWidth - 20}
          height={220}
          yAxisLabel="$"
          chartConfig={{
            backgroundColor: "#000",
            backgroundGradientFrom: "#1e1e1e",
            backgroundGradientTo: "#1e1e1e",
            decimalPlaces: 2,
            color: () => `#00ff00`,
            labelColor: () => `#fff`,
          }}
          style={styles.chart}
        />
      )}
    </ScrollView>
  );
}

/* =================== CRYPTO SCREEN =================== */
function CryptoScreen() {
  const [cryptos, setCryptos] = useState([]);
  const fetchCrypto = async () => {
    try {
      const list = ["BTC", "ETH", "XRP"];
      const results = [];
      for (let c of list) {
        const url = `https://www.alphavantage.co/query?function=CURRENCY_EXCHANGE_RATE&from_currency=${c}&to_currency=USD&apikey=${ALPHA_KEY}`;
        const res = await fetch(url);
        const data = await res.json();
        const price = data["Realtime Currency Exchange Rate"]["5. Exchange Rate"];
        results.push({ symbol: c, price });
      }
      setCryptos(results);
    } catch (e) {
      console.log("Crypto error:", e);
    }
  };
  useEffect(() => {
    fetchCrypto();
  }, []);
  return (
    <SafeAreaView style={styles.container}>
      <Text style={styles.header}>₿ Crypto Prices</Text>
      <FlatList
        data={cryptos}
        keyExtractor={(i) => i.symbol}
        renderItem={({ item }) => (
          <View style={styles.card}>
            <Text style={styles.title}>{item.symbol}</Text>
            <Text style={styles.text}>${parseFloat(item.price).toFixed(2)}</Text>
          </View>
        )}
      />
    </SafeAreaView>
  );
}

/* =================== FOREX SCREEN =================== */
function ForexScreen() {
  const [fx, setFx] = useState([]);
  const pairs = [
    { from: "USD", to: "IDR" },
    { from: "EUR", to: "USD" },
    { from: "JPY", to: "USD" },
  ];
  const fetchFX = async () => {
    try {
      const results = [];
      for (let p of pairs) {
        const url = `https://www.alphavantage.co/query?function=CURRENCY_EXCHANGE_RATE&from_currency=${p.from}&to_currency=${p.to}&apikey=${ALPHA_KEY}`;
        const res = await fetch(url);
        const data = await res.json();
        const rate = data["Realtime Currency Exchange Rate"]["5. Exchange Rate"];
        results.push({ pair: `${p.from}/${p.to}`, rate });
      }
      setFx(results);
    } catch (e) {
      console.log("FX error:", e);
    }
  };
  useEffect(() => {
    fetchFX();
  }, []);
  return (
    <SafeAreaView style={styles.container}>
      <Text style={styles.header}>💱 Forex Rates</Text>
      <FlatList
        data={fx}
        keyExtractor={(i) => i.pair}
        renderItem={({ item }) => (
          <View style={styles.card}>
            <Text style={styles.title}>{item.pair}</Text>
            <Text style={styles.text}>{parseFloat(item.rate).toFixed(2)}</Text>
          </View>
        )}
      />
    </SafeAreaView>
  );
}

/* =================== NEWS SCREEN =================== */
function NewsScreen() {
  const [news, setNews] = useState([]);
  const fetchNews = async () => {
    try {
      const url = `https://newsapi.org/v2/top-headlines?category=business&apiKey=${NEWS_KEY}&language=en`;
      const res = await fetch(url);
      const data = await res.json();
      setNews(data.articles || []);
    } catch (e) {
      console.log("News error:", e);
    }
  };
  useEffect(() => {
    fetchNews();
  }, []);
  return (
    <ScrollView style={styles.container}>
      <Text style={styles.header}>📰 Financial News</Text>
      {news.map((n, i) => (
        <View key={i} style={styles.card}>
          <Text style={styles.title}>{n.title}</Text>
          <Text style={styles.text}>{n.source?.name}</Text>
        </View>
      ))}
    </ScrollView>
  );
}

/* =================== SETTINGS SCREEN =================== */
function SettingsScreen({ toggleTheme, darkMode }) {
  return (
    <SafeAreaView style={styles.container}>
      <Text style={styles.header}>⚙️ Settings</Text>
      <Button title={darkMode ? "Switch to Light Mode" : "Switch to Dark Mode"} onPress={toggleTheme} />
      <Text style={styles.text}>API Keys are set in code. Replace ALPHA_KEY & NEWS_KEY.</Text>
    </SafeAreaView>
  );
}

/* =================== APP MAIN =================== */
export default function App() {
  const [darkMode, setDarkMode] = useState(false);

  return (
    <NavigationContainer theme={darkMode ? DarkTheme : DefaultTheme}>
      <Tab.Navigator
        screenOptions={({ route }) => ({
          tabBarIcon: ({ color, size }) => {
            let icon = "home";
            if (route.name === "Markets") icon = "show-chart";
            else if (route.name === "Stocks") icon = "trending-up";
            else if (route.name === "Crypto") icon = "currency-bitcoin";
            else if (route.name === "Forex") icon = "attach-money";
            else if (route.name === "News") icon = "article";
            else if (route.name === "Settings") icon = "settings";
            return <Icon name={icon} size={size} color={color} />;
          },
        })}
      >
        <Tab.Screen name="Markets" component={MarketScreen} />
        <Tab.Screen name="Stocks" component={StocksScreen} />
        <Tab.Screen name="Crypto" component={CryptoScreen} />
        <Tab.Screen name="Forex" component={ForexScreen} />
        <Tab.Screen name="News" component={NewsScreen} />
        <Tab.Screen name="Settings">
          {() => <SettingsScreen toggleTheme={() => setDarkMode(!darkMode)} darkMode={darkMode} />}
        </Tab.Screen>
      </Tab.Navigator>
    </NavigationContainer>
  );
}
