// App.js - تطبيق أتمتة إنستغرام
import React, { useState, useEffect, useRef } from 'react';
import {
  View,
  Text,
  StyleSheet,
  TouchableOpacity,
  Alert,
  ScrollView,
  ActivityIndicator,
  SafeAreaView,
  StatusBar,
  Share,
  Platform
} from 'react-native';
import { WebView } from 'react-native-webview';
import AsyncStorage from '@react-native-async-storage/async-storage';
import * as FileSystem from 'expo-file-system';
import * as Sharing from 'expo-sharing';

const INSTAGRAM_URL = 'https://www.instagram.com/accounts/login/';
const COOKIES_KEY = '@instagram_cookies';

export default function App() {
  const webViewRef = useRef(null);
  const [isLoggedIn, setIsLoggedIn] = useState(false);
  const [cookies, setCookies] = useState(null);
  const [showWebView, setShowWebView] = useState(false);
  const [loading, setLoading] = useState(false);
  const [status, setStatus] = useState('');
  const [username, setUsername] = useState('');

  useEffect(() => {
    loadSavedCookies();
  }, []);

  const loadSavedCookies = async () => {
    try {
      const savedCookies = await AsyncStorage.getItem(COOKIES_KEY);
      if (savedCookies) {
        setCookies(JSON.parse(savedCookies));
        setIsLoggedIn(true);
        const parsed = JSON.parse(savedCookies);
        extractUsername(parsed);
      }
    } catch (error) {
      console.error('Error loading cookies:', error);
    }
  };

  const extractUsername = (cookiesArray) => {
    const usernameCookie = cookiesArray.find(c => c.name === 'ds_user' || c.name === 'username');
    if (usernameCookie) {
      setUsername(usernameCookie.value);
    }
  };

  const handleWebViewNavigation = async (navState) => {
    const { url } = navState;
    
    // Check if user has successfully logged in
    if (url.includes('instagram.com') && !url.includes('login')) {
      setStatus('تم تسجيل الدخول بنجاح! جاري استخراج الجلسة...');
      setLoading(true);
      
      try {
        // Extract cookies from WebView
        const jsCode = `
          (function() {
            return document.cookie;
          })();
        `;
        
        const cookieString = await webViewRef.current?.injectJavaScript(jsCode);
        
        // Wait a bit for the promise to resolve
        setTimeout(async () => {
          const extractedCookies = await extractCookiesFromWebView();
          if (extractedCookies && extractedCookies.length > 0) {
            await saveCookies(extractedCookies);
            setCookies(extractedCookies);
            setIsLoggedIn(true);
            extractUsername(extractedCookies);
            setStatus('تم حفظ الجلسة بنجاح!');
            setShowWebView(false);
          }
          setLoading(false);
        }, 2000);
      } catch (error) {
        console.error('Error extracting cookies:', error);
        setStatus('فشل في استخراج الجلسة');
        setLoading(false);
      }
    }
  };

  const extractCookiesFromWebView = async () => {
    try {
      const cookieScript = `
        (function() {
          var cookies = document.cookie.split(';');
          var cookieArray = [];
          cookies.forEach(function(cookie) {
            var parts = cookie.trim().split('=');
            if (parts[0] && parts[1]) {
              cookieArray.push({
                name: parts[0].trim(),
                value: parts[1].trim(),
                domain: '.instagram.com',
                path: '/',
                secure: true,
                httpOnly: false
              });
            }
          });
          return JSON.stringify(cookieArray);
        })();
      `;
      
      const result = await webViewRef.current?.injectJavaScript(cookieScript);
      // Parse the result if it comes back as string
      if (result) {
        return JSON.parse(result);
      }
    } catch (error) {
      console.error('Error in extractCookiesFromWebView:', error);
    }
    return [];
  };

  const saveCookies = async (cookiesArray) => {
    try {
      const cookiesJson = JSON.stringify(cookiesArray);
      await AsyncStorage.setItem(COOKIES_KEY, cookiesJson);
      
      // Also save to file
      const fileUri = FileSystem.documentDirectory + 'instagram_session.json';
      await FileSystem.writeAsStringAsync(fileUri, cookiesJson);
      
      Alert.alert('نجاح', 'تم حفظ الجلسة في ملف instagram_session.json');
    } catch (error) {
      console.error('Error saving cookies:', error);
      Alert.alert('خطأ', 'فشل في حفظ الجلسة');
    }
  };

  const exportCookies = async () => {
    if (!cookies) {
      Alert.alert('تنبيه', 'لا توجد جلسة محفوظة');
      return;
    }

    try {
      const fileUri = FileSystem.documentDirectory + 'instagram_session.json';
      const fileInfo = await FileSystem.getInfoAsync(fileUri);
      
      if (fileInfo.exists) {
        await Sharing.shareAsync(fileUri, {
          mimeType: 'application/json',
          dialogTitle: 'تصدير ملف الجلسة',
          UTI: 'public.json'
        });
      } else {
        Alert.alert('خطأ', 'الملف غير موجود');
      }
    } catch (error) {
      console.error('Error sharing file:', error);
      Alert.alert('خطأ', 'فشل في مشاركة الملف');
    }
  };

  const clearSession = async () => {
    Alert.alert(
      'تأكيد',
      'هل أنت متأكد من حذف الجلسة؟',
      [
        { text: 'إلغاء', style: 'cancel' },
        {
          text: 'حذف',
          style: 'destructive',
          onPress: async () => {
            try {
              await AsyncStorage.removeItem(COOKIES_KEY);
              const fileUri = FileSystem.documentDirectory + 'instagram_session.json';
              await FileSystem.deleteAsync(fileUri, { idempotent: true });
              setCookies(null);
              setIsLoggedIn(false);
              setUsername('');
              setStatus('تم حذف الجلسة');
              Alert.alert('تم', 'تم حذف الجلسة بنجاح');
            } catch (error) {
              console.error('Error clearing session:', error);
            }
          }
        }
      ]
    );
  };

  const startLogin = () => {
    setShowWebView(true);
    setStatus('جارٍ فتح صفحة تسجيل الدخول...');
  };

  if (showWebView) {
    return (
      <SafeAreaView style={styles.container}>
        <StatusBar barStyle="dark-content" />
        <View style={styles.header}>
          <TouchableOpacity
            style={styles.backButton}
            onPress={() => setShowWebView(false)}
          >
            <Text style={styles.backButtonText}>✕ إغلاق</Text>
          </TouchableOpacity>
          <Text style={styles.headerTitle}>تسجيل الدخول</Text>
        </View>
        
        {loading && (
          <View style={styles.loadingOverlay}>
            <ActivityIndicator size="large" color="#E1306C" />
            <Text style={styles.loadingText}>{status}</Text>
          </View>
        )}
        
        <WebView
          ref={webViewRef}
          source={{ uri: INSTAGRAM_URL }}
          onNavigationStateChange={handleWebViewNavigation}
          javaScriptEnabled={true}
          domStorageEnabled={true}
          sharedCookiesEnabled={true}
          thirdPartyCookiesEnabled={true}
          startInLoadingState={true}
          renderLoading={() => (
            <View style={styles.webViewLoading}>
              <ActivityIndicator size="large" color="#E1306C" />
              <Text style={styles.loadingText}>جارٍ تحميل إنستغرام...</Text>
            </View>
          )}
          style={styles.webView}
          userAgent="Mozilla/5.0 (Linux; Android 10; SM-G975F) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.120 Mobile Safari/537.36"
        />
      </SafeAreaView>
    );
  }

  return (
    <SafeAreaView style={styles.container}>
      <StatusBar barStyle="light-content" />
      <ScrollView contentContainerStyle={styles.scrollContainer}>
        <View style={styles.mainContent}>
          {/* Header */}
          <View style={styles.titleContainer}>
            <Text style={styles.emoji}>📸</Text>
            <Text style={styles.title}>إنستغرام سشن</Text>
            <Text style={styles.subtitle}>مدير الجلسات الآمن</Text>
          </View>

          {/* Status Card */}
          <View style={styles.statusCard}>
            <View style={[styles.statusDot, isLoggedIn ? styles.statusOnline : styles.statusOffline]} />
            <Text style={styles.statusText}>
              {isLoggedIn ? 'متصل' : 'غير متصل'}
            </Text>
            {username ? (
              <Text style={styles.username}>@{username}</Text>
            ) : null}
          </View>

          {/* Action Buttons */}
          <View style={styles.buttonContainer}>
            {!isLoggedIn ? (
              <TouchableOpacity
                style={[styles.button, styles.loginButton]}
                onPress={startLogin}
              >
                <Text style={styles.buttonText}>🔐 تسجيل الدخول إلى إنستغرام</Text>
              </TouchableOpacity>
            ) : (
              <>
                <TouchableOpacity
                  style={[styles.button, styles.exportButton]}
                  onPress={exportCookies}
                >
                  <Text style={styles.buttonText}>📤 تصدير ملف الجلسة</Text>
                </TouchableOpacity>

                <TouchableOpacity
                  style={[styles.button, styles.reloginButton]}
                  onPress={startLogin}
                >
                  <Text style={styles.buttonText}>🔄 تسجيل دخول جديد</Text>
                </TouchableOpacity>

                <TouchableOpacity
                  style={[styles.button, styles.clearButton]}
                  onPress={clearSession}
                >
                  <Text style={[styles.buttonText, styles.clearButtonText]}>🗑️ حذف الجلسة</Text>
                </TouchableOpacity>
              </>
            )}
          </View>

          {/* Session Info */}
          {cookies && (
            <View style={styles.infoCard}>
              <Text style={styles.infoTitle}>معلومات الجلسة:</Text>
              <Text style={styles.infoText}>عدد الكوكيز: {cookies.length}</Text>
              <Text style={styles.infoText}>الحالة: نشطة ✅</Text>
              <Text style={styles.infoText}>الملف: instagram_session.json</Text>
            </View>
          )}

          {/* Status Message */}
          {status ? (
            <View style={styles.messageBox}>
              <Text style={styles.messageText}>{status}</Text>
            </View>
          ) : null}

          {/* Instructions */}
          <View style={styles.instructionsCard}>
            <Text style={styles.instructionsTitle}>📋 طريقة الاستخدام:</Text>
            <Text style={styles.instructionsText}>
              1. اضغط على زر تسجيل الدخول{'\n'}
              2. سجل دخولك في إنستغرام بأمان{'\n'}
              3. سيتم حفظ الجلسة تلقائياً{'\n'}
              4. يمكنك تصدير ملف الجلسة وقتما تشاء
            </Text>
          </View>

          {/* Warning */}
          <View style={styles.warningCard}>
            <Text style={styles.warningText}>
              ⚠️ تنبيه: هذا التطبيق للأغراض التعليمية فقط. استخدامه مع إنستغرام قد يخالف شروط الخدمة.
            </Text>
          </View>
        </View>
      </ScrollView>
    </SafeAreaView>
  );
}
