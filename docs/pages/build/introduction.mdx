import { Tile } from '../types';
import uuid from 'react-native-uuid';

export function generateBoard(size = 5): Tile[] {
  const tiles: Tile[] = [];

  // create grid
  for (let r = 0; r < size; r++) {
    for (let c = 0; c < size; c++) {
      tiles.push({
        id: uuid.v4() as string,
        row: r,
        col: c,
        type: 'empty',
      });
    }
  }

  // pick a random tile to be the crown
  const crownIdx = Math.floor(Math.random() * tiles.length);
  tiles[crownIdx].type = 'crown';

  // add some traps and items (optional)
  for (let i = 0; i < size; i++) {
    const trapIdx = Math.floor(Math.random() * tiles.length);
    if (tiles[trapIdx].type === 'empty') tiles[trapIdx].type = 'trap';
  }

  return tiles;
}

import React, { useState } from 'react';
import { SafeAreaView, View, Text, StyleSheet, FlatList } from 'react-native';
import { generateBoard } from './src/utils/board';
import { Tile as TileType } from './src/types';
import { Dice } from './src/components/Dice';
import { Tile } from './src/components/Tile';

const BOARD_SIZE = 5;

export default function App() {
  const [tiles, setTiles] = useState<TileType[]>(generateBoard(BOARD_SIZE));

  // single player for demo; extend to multiple players as needed
  const [playerPos, setPlayerPos] = useState({ row: 0, col: 0 });
  const [gold, setGold] = useState(10);
  const [diceValue, setDiceValue] = useState<number | null>(null);

  // find the tile at a given coordinate
  const getTileAt = (row: number, col: number) =>
    tiles.find((t) => t.row === row && t.col === col);

  const rollDice = () => {
    const value = Math.floor(Math.random() * 6) + 1;
    setDiceValue(value);
    movePlayer(value);
  };

  const movePlayer = (steps: number) => {
    // simple linear movement: move right, then down when at edge
    let { row, col } = playerPos;
    for (let i = 0; i < steps; i++) {
      if (col + 1 < BOARD_SIZE) col += 1;
      else if (row + 1 < BOARD_SIZE) { col = 0; row += 1; }
      // otherwise stay in place
    }
    setPlayerPos({ row, col });

    const landedTile = getTileAt(row, col);
    if (!landedTile) return;

    if (landedTile.type === 'trap') {
      alert('You triggered a trap! Lose 2 Gold.');
      setGold((g) => Math.max(g - 2, 0));
    } else if (landedTile.type === 'crown') {
      alert(`ðŸŽ‰ You found the crown and win with ${gold} gold left!`);
    }
  };

  const renderItem = ({ item }: { item: TileType }) => (
    <Tile
      tile={item}
      player={
        item.row === playerPos.row && item.col === playerPos.col
      }
    />
  );

  return (
    <SafeAreaView style={styles.container}>
      <Text style={styles.title}>Quest for the Crown</Text>
      <FlatList
        data={tiles}
        renderItem={renderItem}
        keyExtractor={(t) => t.id}
        numColumns={BOARD_SIZE}
        scrollEnabled={false}
      />
      <View style={styles.info}>
        <Text>Gold: {gold}</Text>
        {diceValue !== null && <Text>Last Roll: {diceValue}</Text>}
      </View>
      <Dice onRoll={rollDice} />
    </SafeAreaView>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1, alignItems: 'center', paddingTop: 20 },
  title: { fontSize: 24, marginBottom: 10 },
  info: { marginVertical: 10 },
});

**EAS Build** is a hosted service for building app binaries for your Expo and React Native projects.

It makes building your apps for distribution simple and easy to automate by providing defaults that work well for Expo and React Native projects out of the box, and by handling your app signing credentials for you (if you wish). It also makes sharing builds with your team easier than ever with [internal distribution](/build/internal-distribution/) (using ad hoc and/or enterprise "universal" provisioning), deeply integrates with EAS Submit for app store submissions, and has first-class support for the [`expo-updates`](/build/updates/) library.

It's designed to work for any native project, whether or not you use Expo and React Native. It's the fastest way to get from `npx create-expo-app` or `npx @react-native-community/cli@latest init` to app stores.
