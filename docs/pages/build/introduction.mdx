  Text,
  View,
  ScrollView,
  Image,
  TouchableOpacity,
  Modal,
  TextInput,
  Alert,

  ActivityIndicator,
  SafeAreaView,
  StatusBar

const db = firebase.firestore();

export default function App() {
  const [news, setNews] = useState([]);
  const [loading, setLoading] = useState(true);
  const [selectedNews, setSelectedNews] = useState(null);
  
  // Matangazo State (Box 1-4)
  const [ads, setAds] = useState({});
  const [showOuterAd, setShowOuterAd] = useState(false);
  const [showInnerAd, setShowInnerAd] = useState(false);

  // Feedback State
  const [feedbackModal, setFeedbackModal] = useState(false);
  const [userMsg, setUserMsg] = useState('');
  const [userName, setUserName] = useState('');

  // Check User Status (Simulated ID or Device ID)
  const [isBlocked, setIsBlocked] = useState(false);
  const [blockMessage, setBlockMessage] = useState('');

  useEffect(() => {
    // A. Vuta Habari Live kutoka Firebase
    const unsubscribeNews = db.collection('news')
      .orderBy('createdAt', 'desc')
      .onSnapshot(snapshot => {
        const newsList = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
        setNews(newsList);
        setLoading(false);
      });

    // B. Vuta Matangazo Live (Box 1, 2, 3, 4)
    const unsubscribeAds = db.collection('advertisements').onSnapshot(snapshot => {
      let adsData = {};
      snapshot.forEach(doc => {
        adsData[doc.id] = doc.data();
      });
      setAds(adsData);

      // Onyesha Tangazo la Nje (Box 2) pindi App inapofunguka kama lipo
      if (adsData['payment_outer'] && adsData['payment_outer'].isActive) {
        setShowOuterAd(true);
      }
    });

    return () => {
      unsubscribeNews();
      unsubscribeAds();
    };
  }, []);

  // Fungua Habari & Onyesha Tangazo la Ndani (Box 3)
  const openNewsDetail = (item) => {
    if (isBlocked) {
      Alert.alert("Akaunti Imefungiwa", blockMessage || "Umemungiwa kutumia huduma hii.");
      return;
    }
    setSelectedNews(item);
    if (ads['payment_inner'] && ads['payment_inner'].isActive) {
      setShowInnerAd(true);
    }
  };

  // Tuma Maoni
  const sendFeedback = async () => {
    if (!userMsg.trim()) return Alert.alert("Kosa", "Tafadhali andika maoni yako.");
    try {
      await db.collection('feedbacks').add({
        userName: userName || 'Msomaji',
        message: userMsg,
        createdAt: new Date().toISOString()
      });
      Alert.alert("Asante!", "Maoni yako yamefika kwa Mwandishi Kavishe.");
      setUserMsg('');
      setFeedbackModal(false);
    } catch (e) {
      Alert.alert("Kosa", "Imeshindikana kutuma maoni.");
    }
  };

  return (
    <SafeAreaView style={styles.container}>
      <StatusBar barStyle="light-content" backgroundColor="#0f172a" />

      {/* HEADER BAR */}
      <View style={styles.header}>
        <Text style={styles.headerTitle}>KAVISHE <Text style={{color: '#3b82f6'}}>NEWS ROOM</Text></Text>
        <TouchableOpacity onPress={() => setFeedbackModal(true)} style={styles.feedbackBtn}>
          <Text style={styles.feedbackBtnText}>💬 Maoni</Text>
        </TouchableOpacity>
      </View>

      {/* BANNER PROMO YA APP ZANGU (BOX 4 - DIRECT APK LINK) */}
      {ads['my_apps_apk'] && ads['my_apps_apk'].isActive && (
        <TouchableOpacity 
          style={styles.box4Banner}
          onPress={() => ads['my_apps_apk'].targetUrl && Linking.openURL(ads['my_apps_apk'].targetUrl)}
        >
          <Image source={{ uri: ads['my_apps_apk'].bannerUrl }} style={styles.box4Icon} />
          <View style={{flex: 1, marginLeft: 10}}>
            <Text style={styles.box4Title}>{ads['my_apps_apk'].title}</Text>
            <Text style={styles.box4Sub}>Bonyeza hapa kupakua App hii moja kwa moja!</Text>
          </View>
          <View style={styles.downloadBtn}>
            <Text style={styles.downloadBtnText}>Pakua APK</Text>
          </View>
        </TouchableOpacity>
      )}

      {/* ORODHA YA HABARI (NEWS FEED) */}
      {loading ? (
        <View style={styles.centerLoading}>
          <ActivityIndicator size="large" color="#2563eb" />
          <Text style={{marginTop: 10, color: '#64748b', fontSize: 12}}>Mwandishi Kavishe Anapakia Habari...</Text>
        </View>
      ) : (
        <ScrollView style={styles.feedContainer} showsVerticalScrollIndicator={false}>
          {news.map((item, index) => (
            <React.Fragment key={item.id}>
              
              {/* CARD YA HABARI */}
              <TouchableOpacity style={styles.newsCard} activeOpacity={0.85} onPress={() => openNewsDetail(item)}>
                
                {/* CONTAINER YA PICHA SAMA NA WATERMARK YA KAVISHE */}
                <View style={styles.imageWrapper}>
                  <Image source={{ uri: item.imageUrl }} style={styles.newsImage} />
                  
                  {/* WATERMARK YA KAVISHE (KAM MILARD AYO) */}
                  <View style={styles.watermarkContainer}>
                    <Text style={styles.watermarkText}>KAVISHE NEWS</Text>
                  </View>
                </View>

                <View style={styles.newsContent}>
                  <Text style={styles.categoryBadge}>{item.category || 'Trending'}</Text>
                  <Text style={styles.newsTitle}>{item.title}</Text>
                  <Text style={styles.newsTagline}>{item.authorTagline}</Text>
                </View>
              </TouchableOpacity>

              {/* BOX 1: TANGAZO LA BIASHARA (Inajitokeza baadhi ya habari) */}
              {index === 1 && ads['business_banner'] && ads['business_banner'].isActive && (
                <TouchableOpacity 
                  style={styles.businessAdCard}
                  onPress={() => ads['business_banner'].targetUrl && Linking.openURL(ads['business_banner'].targetUrl)}
                >
                  <Text style={styles.adLabel}>TANGAZO LA BIASHARA</Text>
                  <Image source={{ uri: ads['business_banner'].bannerUrl }} style={styles.adImage} />
                  <Text style={styles.adTitle}>{ads['business_banner'].title}</Text>
                </TouchableOpacity>
              )}

            </React.Fragment>
          ))}
        </ScrollView>
      )}

      {/* MODAL 1: SOMA HABARI KAMILI */}
      <Modal visible={!!selectedNews} animationType="slide" onRequestClose={() => setSelectedNews(null)}>
        {selectedNews && (
          <SafeAreaView style={{flex: 1, backgroundColor: '#fff'}}>
            <ScrollView style={{flex: 1}}>
              <View style={styles.imageWrapperDetail}>
                <Image source={{ uri: selectedNews.imageUrl }} style={styles.detailImage} />
                
                {/* WATERMARK YENYE GUO KISASA (DETAIL SCREEN) */}
                <View style={styles.watermarkContainerDetail}>
                  <Text style={styles.watermarkTextDetail}>KAVISHE NEWS ROOM</Text>
                </View>
              </View>

              <View style={{p: 16, padding: 16}}>
                <Text style={styles.categoryBadge}>{selectedNews.category}</Text>
                <Text style={styles.detailTitle}>{selectedNews.title}</Text>
                <Text style={styles.detailAuthor}>{selectedNews.authorTagline}</Text>
                <View style={styles.divider} />
                <Text style={styles.detailBody}>{selectedNews.content}</Text>
              </View>
            </ScrollView>
            
            <TouchableOpacity style={styles.closeDetailBtn} onPress={() => setSelectedNews(null)}>
              <Text style={{color: '#fff', fontWeight: 'bold'}}>Rudi Kwenye Habari</Text>
            </TouchableOpacity>
          </SafeAreaView>
        )}
      </Modal>

      {/* MODAL 2: TANGAZO LA NJE (BOX 2 - ON LAUNCH) */}
      <Modal visible={showOuterAd} transparent animationType="fade">
        <View style={styles.modalOverlay}>
          <View style={styles.adPopUpBox}>
            <Text style={styles.popUpHeader}>TAARIFA MAALUM</Text>
            {ads['payment_outer']?.bannerUrl && (
              <Image source={{ uri: ads['payment_outer'].bannerUrl }} style={styles.popUpImage} />
            )}
            <Text style={styles.popUpTitle}>{ads['payment_outer']?.title}</Text>
            <TouchableOpacity 
              style={styles.popUpActionBtn}
              onPress={() => {
                if (ads['payment_outer']?.targetUrl) Linking.openURL(ads['payment_outer'].targetUrl);
                setShowOuterAd(false);
              }}
            >
              <Text style={styles.popUpActionText}>Kamilisha / Endelea</Text>
            </TouchableOpacity>
            <TouchableOpacity onPress={() => setShowOuterAd(false)} style={{marginTop: 10}}>
              <Text style={{color: '#94a3b8', fontSize: 12}}>Funga Tangazo</Text>
            </TouchableOpacity>
          </View>
        </View>
      </Modal>

      {/* MODAL 3: TANGAZO LA NDANI (BOX 3 - POPUP WHEN READING) */}
      <Modal visible={showInnerAd} transparent animationType="slide">
        <View style={styles.modalOverlay}>
          <View style={[styles.adPopUpBox, {borderColor: '#a855f7', borderWidth: 2}]}>
            <Text style={[styles.popUpHeader, {color: '#a855f7'}]}>HUDUMA YA PREMIUM</Text>
            {ads['payment_inner']?.bannerUrl && (
              <Image source={{ uri: ads['payment_inner'].bannerUrl }} style={styles.popUpImage} />
            )}
            <Text style={styles.popUpTitle}>{ads['payment_inner']?.title}</Text>
            <TouchableOpacity 
              style={[styles.popUpActionBtn, {backgroundColor: '#9333ea'}]}
              onPress={() => {
                if (ads['payment_inner']?.targetUrl) Linking.openURL(ads['payment_inner'].targetUrl);
                setShowInnerAd(false);
              }}
            >
              <Text style={styles.popUpActionText}>Lipia / Angalia</Text>
            </TouchableOpacity>
            <TouchableOpacity onPress={() => setShowInnerAd(false)} style={{marginTop: 10}}>
              <Text style={{color: '#94a3b8', fontSize: 12}}>Endelea Kusoma Habari</Text>
            </TouchableOpacity>
          </View>
        </View>
      </Modal>

      {/* MODAL 4: SEND FEEDBACK */}
      <Modal visible={feedbackModal} animationType="slide" transparent>
        <View style={styles.modalOverlay}>
          <View style={styles.feedbackBox}>
            <Text style={styles.feedbackHeader}>Tuma Maoni kwa Mwandishi Kavishe</Text>
            <TextInput 
              placeholder="Jina Lako (Sio lazima)" 
              style={styles.input}
              value={userName}
              onChangeText={setUserName}
            />
            <TextInput 
              placeholder="Andika maoni au ushauri wako hapa..." 
              style={[styles.input, {height: 80}]} 
              multiline
              value={userMsg}
              onChangeText={setUserMsg}
            />
            <View style={{flexDirection: 'row', gap: 10}}>
              <TouchableOpacity onPress={() => setFeedbackModal(false)} style={[styles.modalBtn, {backgroundColor: '#64748b'}]}>
                <Text style={{color: '#fff', fontWeight: 'bold'}}>Ghairi</Text>
              </TouchableOpacity>
              <TouchableOpacity onPress={sendFeedback} style={[styles.modalBtn, {backgroundColor: '#2563eb'}]}>
                <Text style={{color: '#fff', fontWeight: 'bold'}}>Tuma</Text>
              </TouchableOpacity>
            </View>
          </View>
        </View>
      </Modal>

    </SafeAreaView>
  );
}
