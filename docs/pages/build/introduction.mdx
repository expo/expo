      style={{ background: s.tintBg, color: s.tint }}

  );

export default function App() {
  const [role, setRole] = useState(null);
  const [userScreen, setUserScreen] = useState("home");
  const [selectedProduct, setSelectedProduct] = useState(null);
  const [selectedSectorId, setSelectedSectorId] = useState(null);
  const [cart, setCart] = useState([]);
  const [flowStep, setFlowStep] = useState("browse"); // browse | detail | cart | checkout | success

  const handleAddToCart = (product, sectorId) => {
    setCart((c) => [...c, { product, sectorId }]);
    setFlowStep("cart");
  };

  const handleSelectProduct = (product, sectorId) => {
    setSelectedProduct(product);
    setSelectedSectorId(sectorId);
    setFlowStep("detail");
  };

  const resetToHome = () => {
    setFlowStep("browse");
    setUserScreen("home");
    setSelectedProduct(null);
  };

  if (!role) {
    return (
      <div dir="rtl" className="max-w-md mx-auto min-h-screen" style={{ fontFamily: "'IBM Plex Sans Arabic', 'Noto Sans Arabic', sans-serif" }}>
        <RoleSelect onSelect={setRole} />
      </div>
    );
  }

  return (
    <div dir="rtl" className="max-w-md mx-auto min-h-screen relative" style={{ fontFamily: "'IBM Plex Sans Arabic', 'Noto Sans Arabic', sans-serif", background: "#FAF7F1" }}>
      {role === "partner" && <PartnerDashboard onSwitchRole={() => setRole(null)} />}
      {role === "admin" && <AdminDashboard onSwitchRole={() => setRole(null)} />}

      {role === "user" && (
        <>
          {flowStep === "detail" && selectedProduct && (
            <ProductDetail
              product={selectedProduct}
              sectorId={selectedSectorId}
              onBack={() => setFlowStep("browse")}
              onAddToCart={handleAddToCart}
            />
          )}

          {flowStep === "cart" && (
            <CartView
              cart={cart}
              onBack={() => setFlowStep("browse")}
              onCheckout={() => setFlowStep("checkout")}
              onRemove={(i) => setCart((c) => c.filter((_, idx) => idx !== i))}
            />
          )}

          {flowStep === "checkout" && (
            <CheckoutView onBack={() => setFlowStep("cart")} onComplete={() => { setCart([]); setFlowStep("success"); }} />
          )}

          {flowStep === "success" && <SuccessView onDone={resetToHome} />}

          {flowStep === "browse" && (
            <>
              {userScreen === "home" && (
                <UserHome
                  onSelectProduct={handleSelectProduct}
                  cartCount={cart.length}
                  onOpenCart={() => setFlowStep("cart")}
                  onOpenNotifications={() => setUserScreen("notifications")}
                  onOpenProfile={() => setUserScreen("profile")}
                />
              )}
              {userScreen === "map" && <MapView onBack={() => setUserScreen("home")} />}
              {userScreen === "notifications" && <NotificationsView onBack={() => setUserScreen("home")} />}
              {userScreen === "profile" && (
                <div>
                  <ProfileView onBack={() => setUserScreen("home")} />
                  <button
                    onClick={() => setRole(null)}
                    className="fixed bottom-20 left-1/2 -translate-x-1/2 max-w-md text-[11px] font-bold px-4 py-2 rounded-full text-white z-40"
                    style={{ background: "#1B4F4A" }}
                  >
                    تبديل نوع الحساب (للعرض)
                  </button>
                </div>
              )}
              <BottomNav
                active={userScreen === "cart" ? "cart" : userScreen}
                onChange={(id) => (id === "cart" ? setFlowStep("cart") : setUserScreen(id))}
                cartCount={cart.length}
              />
            </>
          )}
        </>
      )}
    </div>
  );
}
