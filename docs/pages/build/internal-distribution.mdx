import React from "react";
import { StyleSheet, Text, View, TouchableOpacity, Linking, Alert } from "react-native";

export default function App() {
  const gameGuardianURL = "gameguardian://";
  const cloneAppURL = "cloneapp://";

  const openLink = (url) => {
    Linking.canOpenURL(url).then((supported) => {
      if (supported) {
        Linking.openURL(url);
      } else {
        Alert.alert("ไม่พบแอป", "กรุณาติดตั้งแอปที่เกี่ยวข้องก่อน");
      }
    });
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>BanLai Guardian</Text>
      <TouchableOpacity
        style={[styles.button, { backgroundColor: "#22c55e" }]}
        onPress={() => openLink(gameGuardianURL)}
      >
        <Text style={styles.buttonText}>เปิด Game Guardian</Text>
      </TouchableOpacity>

      <TouchableOpacity
        style={[styles.button, { backgroundColor: "#2563eb" }]}
        onPress={() => openLink(cloneAppURL)}
      >
        <Text style={styles.buttonText}>เข้าสู่ Clone App</Text>
      </TouchableOpacity>

      <TouchableOpacity
        style={[styles.button, { backgroundColor: "#eab308" }]}
        onPress={() =>
          Alert.alert(
            "คู่มือการใช้งาน",
            "1. ติดตั้ง Game Guardian และ Clone App\n2. เปิดแอปนี้เพื่อเข้าถึงง่าย\n3. ใช้งาน Game Guardian ผ่าน Clone App โดยไม่ต้องรูท"
          )
        }
      >
        <Text style={styles.buttonText}>คู่มือการใช้งาน</Text>
      </TouchableOpacity>
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: "#000",
    alignItems: "center",
    justifyContent: "center",
    paddingHorizontal: 20,
  },
  title: {
    fontSize: 32,
    fontWeight: "bold",
    color: "#fff",
    marginBottom: 40,
  },
  button: {
    width: "100%",
    paddingVertical: 18,
    borderRadius: 12,
    marginBottom: 20,
    alignItems: "center",
  },
  buttonText: {
    color: "#fff",
    fontSize: 18,
    fontWeight: "600",
  },
});
