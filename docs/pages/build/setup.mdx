from kivy.app import App
from kivy.uix.boxlayout import BoxLayout
from kivy.uix.label import Label
from kivy.uix.textinput import TextInput
from kivy.uix.button import Button

class GreetingApp(App):
    def build(self):
        self.layout = BoxLayout(orientation='vertical', padding=20, spacing=10)

        self.label = Label(text="তোমার নাম লিখো:", font_size=24)
        self.layout.add_widget(self.label)

        self.text_input = TextInput(multiline=False, font_size=24)
        self.layout.add_widget(self.text_input)

        self.greet_button = Button(text="শুভেচ্ছা দেখাও", font_size=24)
        self.greet_button.bind(on_press=self.show_greeting)
        self.layout.add_widget(self.greet_button)

        self.greeting_label = Label(text="", font_size=24)
        self.layout.add_widget(self.greeting_label)

        return self.layout

    def show_greeting(self, instance):
        name = self.text_input.text.strip()
        if name:
            self.greeting_label.text = f"শুভেচ্ছা, {name}!"
        else:
            self.greeting_label.text = "তোমার নাম লিখো দয়া করে।"

if __name__ == "__main__":
    GreetingApp().run()
