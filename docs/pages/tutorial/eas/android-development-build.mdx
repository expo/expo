import arabic_reshaper
from bidi.algorithm import get_display
from kivymd.app import MDApp
from kivymd.uix.boxlayout import MDBoxLayout
from kivymd.uix.button import MDRaisedButton, MDFlatButton
from kivymd.uix.label import MDLabel
from kivymd.uix.textfield import MDTextField
from kivymd.uix.toolbar import MDTopAppBar
from kivymd.uix.card import MDCard

def ar(text):
    """دالة معالجة النصوص العربية لضمان ظهورها بشكل صحيح وسليم على الهاتف ومن اليمين لليسار"""
    return get_display(arabic_reshaper.reshape(text))

class StudentLoginForm(MDBoxLayout):
    """واجهة دخول الطلاب وأولياء الأمور - تطلب رقم الهاتف فقط"""
    def __init__(self, login_callback, **kwargs):
        super().__init__(orientation='vertical', spacing="20dp", **kwargs)
        
        self.add_widget(MDLabel(text=ar("تسجيل دخول الطلاب وأولياء الأمور"), halign="center", font_style="H6"))
        
        # حقل رقم الهاتف فقط للطالب أو ولي الأمر
        self.phone_input = MDTextField(
            hint_text=ar("أدخل رقم الهاتف المسجل فقط"), 
            icon_right="phone", 
            mode="round"
        )
        
        btn_submit = MDRaisedButton(
            text=ar("دخول حساب الطالب / ولي الأمر"), 
            pos_hint={"center_x": 0.5}, 
            size_hint_x=0.9,
            on_press=lambda inst: login_callback("طالب", self.phone_input.text, None)
        )
        
        self.add_widget(self.phone_input)
        self.add_widget(btn_submit)

class TeacherLoginForm(MDBoxLayout):
    """واجهة دخول الشيخ - تطلب الاسم (الشيخ أحمد) والكود 125"""
    def __init__(self, login_callback, **kwargs):
        super().__init__(orientation='vertical', spacing="15dp", **kwargs)
        
        self.add_widget(MDLabel(text=ar("تسجيل دخول الشيوخ والمعلمين"), halign="center", font_style="H6"))
        
        # حقل الاسم ومثبت فيه تلقائياً "الشيخ أحمد" كما طلبت
        self.name_input = MDTextField(
            text=ar("الشيخ أحمد"),
            hint_text=ar("اسم الشيخ بالكامل"), 
            icon_right="account", 
            mode="round"
        )
        
        # حقل الكود المكون من (125)
        self.code_input = MDTextField(
            hint_text=ar("أدخل كود الشيخ المكون من (125)"), 
            icon_right="lock", 
            password=True, 
            mode="round"
        )
        
        btn_submit = MDRaisedButton(
            text=ar("دخول لوحة تحكم الشيخ"), 
            pos_hint={"center_x": 0.5}, 
            size_hint_x=0.9,
            on_press=lambda inst: login_callback("شيخ", self.name_input.text, self.code_input.text)
        )
        
        self.add_widget(self.name_input)
        self.add_widget(self.code_input)
        self.add_widget(btn_submit)

class NoorQuranApp(MDApp):
    def build(self):
        # الهوية البصرية الرسمية لمركز نور القرآن (الأخضر الإسلامي والذهبي)
        self.theme_cls.primary_palette = "Green"
        self.theme_cls.primary_hue = "800"
        self.theme_cls.accent_palette = "Amber"
        
        main_layout = MDBoxLayout(orientation='vertical', bg_color=self.theme_cls.bg_normal)
        
        # شريط العنوان العلوي
        toolbar = MDTopAppBar(title=ar("مركز نور القرآن"), anchor_title="center", elevation=4)
        main_layout.add_widget(toolbar)
        
        # كارت توثيق المبرمج ياسين محمد أعلى الواجهة بشكل متوسط وبارز
        header_card = MDCard(orientation='vertical', padding="10dp", size_hint_y=None, height="90dp", elevation=1)
        designer_text = MDLabel(
            text=ar("من تصميم وإنشاء النظام المبرمج ياسين محمد"), 
            halign="center", 
            font_style="Subtitle1", 
            theme_text_color="Custom", 
            text_color=self.theme_cls.primary_color
        )
        welcome_text = MDLabel(text=ar("مرحباً بكم في مركز نور القرآن"), halign="center", font_style="Body2", theme_text_color="Secondary")
        header_card.add_widget(designer_text)
        header_card.add_widget(welcome_text)
        main_layout.add_widget(header_card)
        
        # أزرار التبديل العلوية (دخول الطالب / دخول الشيخ)
        tabs_layout = MDBoxLayout(orientation='horizontal', size_hint_y=None, height="55dp", padding="10dp", spacing="10dp")
        self.btn_student_tab = MDRaisedButton(text=ar("دخول الطلاب وأولياء الأمور"), size_hint_x=0.5, on_press=self.switch_to_student)
        self.btn_teacher_tab = MDFlatButton(text=ar("دخول الشيخ والأقسام"), size_hint_x=0.5, on_press=self.switch_to_teacher)
        tabs_layout.add_widget(self.btn_student_tab)
        tabs_layout.add_widget(self.btn_teacher_tab)
        main_layout.add_widget(tabs_layout)
        
        # حاوية النوافذ الديناميكية
        self.container = MDBoxLayout(padding="20dp")
        main_layout.add_widget(self.container)
        
        # تحضير واجهات الدخول مسبقاً لمنع البطء عند التنقل
        self.student_view = StudentLoginForm(login_callback=self.handle_login_process)
        self.teacher_view = TeacherLoginForm(login_callback=self.handle_login_process)
        
        # العرض الافتراضي هو واجهة الطالب وولي الأمر برقم الهاتف
        self.switch_to_student(None)
        return main_layout

    def switch_to_student(self, instance):
        """الانتقال لواجهة الطالب برقم الهاتف وتحديث الألوان"""
        self.btn_student_tab.md_bg_color = self.theme_cls.primary_color
        self.btn_student_tab.text_color = [1, 1, 1, 1]
        self.btn_teacher_tab.md_bg_color = [0, 0, 0, 0]
        self.btn_teacher_tab.text_color = self.theme_cls.primary_color
        self.container.clear_widgets()
        self.container.add_widget(self.student_view)

    def switch_to_teacher(self, instance):
        """الانتقال لواجهة الشيخ بالاسم المكتوب والكود (125) وتحديث الألوان"""
        self.btn_teacher_tab.md_bg_color = self.theme_cls.primary_color
        self.btn_teacher_tab.text_color = [1, 1, 1, 1]
        self.btn_student_tab.md_bg_color = [0, 0, 0, 0]
        self.btn_student_tab.text_color = self.theme_cls.primary_color
        self.container.clear_widgets()
        self.container.add_widget(self.teacher_view)

    def handle_login_process(self, role, identity, credential):
        """دالة معالجة واستقبال المدخلات النظيفة من الواجهتين من قبل المبرمج ياسين"""
        if role == "طالب":
            print(f"[نظام المبرمج ياسين] دخول طالب/ولي أمر برقم الهاتف: {identity}")
        elif role == "شيخ":
            print(f"[نظام المبرمج ياسين] دخول الشيخ باسم: {identity} | الكود الرقمي: {credential}")

if __name__ == "__main__":
    NoorQuranApp().run()
