import React, { useState, useEffect } from 'react';
import { View, Text, Button, Image, TouchableOpacity, Alert, StyleSheet } from 'react-native';

const SERVER_URL = 'http://your-server-url.com'; // replace with real IP/server

export default function App() {
  const [coins, setCoins] = useState(100);
  const [spinning, setSpinning] = useState(false);

  const handleSpin = async () => {
    setSpinning(true);

      const response = await fetch(`${SERVER_URL}/spin`, { method: 'POST' });
      const data = await response.json();

      Alert.alert(`ðŸŽ‰ You won ${data.reward} coins!`);
    } catch (err) {
      Alert.alert('Error spinning wheel.');

  };

      </TouchableOpacity>

        style={styles.telegramButton}
        onPress={() => {
          Alert.alert('Redirecting to Telegram...');
        }}

      </TouchableOpacity>

// Simple coin reward logic with RTP
function getSpinReward() {
  const rtp = 90; // 90% RTP
  const probabilities = [
    { reward: 1000, chance: 0.0000005 }, // Grand prize
    { reward: 10, chance: 0.2 },
    { reward: 5, chance: 0.3 },
    { reward: 1, chance: 0.5 },
    { reward: 0, chance: 0.1 },
  ];

  let rand = Math.random();
  let sum = 0;
  for (let item of probabilities) {
    sum += item.chance;
    if (rand < sum) return item.reward;
  }
  return 1;
}

app.post('/spin', (req, res) => {
  const reward = getSpinReward();
  res.json({ reward });
});
