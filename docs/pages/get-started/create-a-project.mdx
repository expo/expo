// -------- Composants Réutilisables --------

// Affichage d'une note avec des étoiles
const StarRating = ({ rating }) => {
  const fullStars = Math.floor(rating);
  return (
    <Text style={{ color: "#FBBF24", fontSize: 13 }}>
      {"★".repeat(fullStars)}{"☆".repeat(5 - fullStars)} {rating.toFixed(1)}
    </Text>
  );
};

// Badges pour afficher des informations
const Badge = ({ text, color, textColor }) => {
  return (
    <View
      style={{
        backgroundColor: color || C.blueLight,
        borderRadius: 20,
        paddingHorizontal: 10,
        paddingVertical: 5,
        marginRight: 5,
        marginBottom: 5,
      }}
    >
      <Text
        style={{
          color: textColor || C.blue,
          fontSize: 12,
          fontWeight: "600",
        }}
      >
        {text}
      </Text>
    </View>
  );
};

// Bouton réutilisable
const Btn = ({ title, onPress, outline }) => {
  return (
    <TouchableOpacity
      style={[
        {
          paddingVertical: 14,
          borderRadius: 14,
          alignItems: "center",
          backgroundColor: outline ? C.white : C.blue,
          borderWidth: outline ? 2 : 0,
          borderColor: C.blue,
        },
      ]}
      onPress={onPress}
    >
      <Text style={{ color: outline ? C.blue : C.white, fontWeight: "700" }}>
        {title}
      </Text>
    </TouchableOpacity>
  );
};

// Composant d'une carte de trajet
const TripCard = ({ trip }) => {
  return (
    <View style={styles.card}>

          <Text style={{ fontWeight: "700", fontSize: 15 }}>{trip.driver.name}</Text>

        </View>

        {/* Prix */}
        <View style={{ alignItems: "flex-end" }}>
          <Text style={{ fontSize: 18, fontWeight: "700", color: C.blue }}>
            {trip.price}$
          </Text>
          <Text style={{ fontSize: 12, color: C.grayMid }}>/place</Text>
        </View>
      </View>

      {/* Détails du trajet */}
      <View style={[styles.row, { marginTop: 12 }]}>
        <Badge text={`${trip.date} à ${trip.time}`} />
        <Badge text={`Durée : ${trip.duration}`} color={C.greenLight} textColor={C.green} />
        <Badge text={`${trip.seats} places`} color={C.orangeLight} />
      </View>
    </View>
  );
};
