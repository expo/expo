// Note: This is a basic starter script. You can expand physics, animations, UI, etc.

using UnityEngine;

public class MotorcycleController : MonoBehaviour
{
    public float speed = 30f;          // Forward speed
    public float turnSpeed = 100f;     // Turning sensitivity
    public float tiltAmount = 15f;     // Leaning animation

    private Rigidbody rb;

    void Start()
    {
        rb = GetComponent<Rigidbody>();
        rb.centerOfMass = new Vector3(0, -0.5f, 0); // More stability
    }

    void FixedUpdate()
    {
        // Forward movement
        float moveInput = Input.GetAxis("Vertical");
        rb.AddForce(transform.forward * moveInput * speed);

        // Turning
        float turnInput = Input.GetAxis("Horizontal");
        transform.Rotate(0, turnInput * turnSpeed * Time.fixedDeltaTime, 0);

        // Leaning effect
        float tilt = -turnInput * tiltAmount;
        Vector3 currentRotation = transform.localEulerAngles;
        transform.localEulerAngles = new Vector3(currentRotation.x, currentRotation.y, tilt);
    }
}
