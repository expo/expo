    <script>
        document.addEventListener('DOMContentLoaded', function() {
            // العناصر الرئيسية
            const addTaskForm = document.getElementById('addTaskForm');
            const taskList = document.getElementById('taskList');
            const filterBtns = document.querySelectorAll('.filter-btn');
            
            // تحميل المهام من localStorage عند بدء التطبيق
            let tasks = JSON.parse(localStorage.getItem('tasks')) || [];
            
            // عرض المهام
            function renderTasks(filter = 'all') {
                taskList.innerHTML = '';
                
                if (tasks.length === 0) {
                    taskList.innerHTML = `
                        <div class="empty-state">
                            <i class="fas fa-clipboard-list"></i>
                            <h3>لا توجد مهام حتى الآن</h3>
                            <p>ابدأ بإضافة مهمة جديدة باستخدام النموذج على اليسار</p>
                        </div>
                    `;
                    updateStats();
                    return;
                }
                
                let filteredTasks = tasks;
                if (filter !== 'all') {
                    if (filter === 'high') {
                        filteredTasks = tasks.filter(task => task.priority === 'high');
                    } else {
                        filteredTasks = tasks.filter(task => task.category === filter);
                    }
                }
                
                if (filteredTasks.length === 0) {
                    taskList.innerHTML = `
                        <div class="empty-state">
                            <i class="fas fa-search"></i>
                            <h3>لا توجد مهام تطابق التصفية</h3>
                            <p>جرب استخدام تصفية مختلفة أو أضف مهام جديدة</p>
                        </div>
                    `;
                    return;
                }
                
                filteredTasks.forEach((task, index) => {
                    const li = document.createElement('li');
                    li.className = `task-item ${task.priority}-priority`;
                    
                    const priorityText = task.priority === 'high' ? 'عالي' : task.priority === 'medium' ? 'متوسط' : 'منخفض';
                    const priorityClass = task.priority === 'high' ? 'priority-high' : task.priority === 'medium' ? 'priority-medium' : 'priority-low';
                    
                    li.innerHTML = `
                        <div class="task-info">
                            <h3>${task.title}</h3>
                            ${task.description ? `<p>${task.description}</p>` : ''}
                            <div class="task-meta">
                                <span><i class="fas fa-tag"></i> <span class="category-badge">${task.category}</span></span>
                                <span><i class="fas fa-exclamation-circle"></i> <span class="priority-badge ${priorityClass}">${priorityText}</span></span>
                                ${task.dueDate ? `<span><i class="fas fa-calendar-day"></i> ${formatDate(task.dueDate)}</span>` : ''}
                            </div>
                        </div>
                        <div class="task-actions">
                            <button class="btn-complete" title="إكمال المهمة"><i class="fas fa-check"></i></button>
                            <button class="btn-edit" title="تعديل المهمة"><i class="fas fa-edit"></i></button>
                            <button class="btn-delete" title="حذف المهمة"><i class="fas fa-trash"></i></button>
                        </div>
                    `;
                    
                    li.querySelector('.btn-complete').addEventListener('click', () => completeTask(index));
                    li.querySelector('.btn-edit').addEventListener('click', () => editTask(index));
                    li.querySelector('.btn-delete').addEventListener('click', () => deleteTask(index));
                    
                    taskList.appendChild(li);
                });
                
                updateStats();
            }
            
            // تحديث الإحصائيات
            function updateStats() {
                document.getElementById('totalTasks').textContent = tasks.length;
                document.getElementById('completedTasks').textContent = tasks.filter(task => task.completed).length;
                document.getElementById('pendingTasks').textContent = tasks.filter(task => !task.completed).length;
                document.getElementById('highPriorityTasks').textContent = tasks.filter(task => task.priority === 'high').length;
            }
            
            // تنسيق التاريخ
            function formatDate(dateString) {
                if (!dateString) return '';
                const options = { year: 'numeric', month: 'long', day: 'numeric' };
                return new Date(dateString).toLocaleDateString('ar-EG', options);
            }
            
            // إضافة مهمة جديدة
            addTaskForm.addEventListener('submit', function(e) {
                e.preventDefault();
                
                const newTask = {
                    title: document.getElementById('taskTitle').value,
                    description: document.getElementById('taskDescription').value,
                    category: document.getElementById('taskCategory').value,
                    priority: document.getElementById('taskPriority').value,
                    dueDate: document.getElementById('taskDueDate').value,
                    completed: false,
                    createdAt: new Date()
                };
                
                tasks.push(newTask);
                localStorage.setItem('tasks', JSON.stringify(tasks));
                
                addTaskForm.reset();
                renderTasks();
                
                alert('تم إضافة المهمة بنجاح!');
            });
            
            // إكمال المهمة
            function completeTask(index) {
                tasks[index].completed = true;
                localStorage.setItem('tasks', JSON.stringify(tasks));
                renderTasks();
                alert('تهانينا! لقد أكملت المهمة بنجاح.');
            }
            
            // تعديل المهمة
            function editTask(index) {
                const task = tasks[index];
                
                document.getElementById('taskTitle').value = task.title;
                document.getElementById('taskDescription').value = task.description;
                document.getElementById('taskCategory').value = task.category;
                document.getElementById('taskPriority').value = task.priority;
                document.getElementById('taskDueDate').value = task.dueDate;
                
                tasks.splice(index, 1);
                localStorage.setItem('tasks', JSON.stringify(tasks));
                renderTasks();
                
                alert('يمكنك الآن تعديل المهمة في نموذج الإضافة');
            }
            
            // حذف المهمة
            function deleteTask(index) {
                if (confirm('هل أنت متأكد من حذف هذه المهمة؟')) {
                    tasks.splice(index, 1);
                    localStorage.setItem('tasks', JSON.stringify(tasks));
                    renderTasks();
                    alert('تم حذف المهمة بنجاح');
                }
            }
            
            // تصفية المهام
            filterBtns.forEach(btn => {
                btn.addEventListener('click', function() {
                    filterBtns.forEach(b => b.classList.remove('active'));
                    this.classList.add('active');
                    
                    const filter = this.getAttribute('data-filter');
                    renderTasks(filter);
                });
            });
            
            // التهيئة الأولية
            renderTasks();
        });
    </script>
