    payouts:       'Payouts <em>Driver Settlements</em>',
    promos:        'Promos & Referrals <em>Campaigns</em>',

    fares:         'Fare Configuration <em>Pricing & Surge Periods</em>',
    reports:       'Reports <em>Analytics & Export</em>',
    settings:      'Settings <em>Platform Config</em>',
  };

  /**
   * Navigate to a page
   * @param {string} id   - page key matching PAGE_TITLES & id="page-{id}"
   * @param {Element|null} clickedEl - the nav-item that was clicked
   */
  function showPage(id, clickedEl) {
    // Hide all pages
    document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
    // Show target page
    const pg = document.getElementById('page-' + id);
    if (pg) pg.classList.add('active');
    // Update nav active state
    document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
    if (clickedEl) clickedEl.classList.add('active');
    // Update title
    document.getElementById('pageTitle').innerHTML = PAGE_TITLES[id] || id;
  }

  /* ── Modal helpers ── */
  function openModal(id)  { document.getElementById(id).classList.add('open');    }
  function closeModal(id) { document.getElementById(id).classList.remove('open'); }

  /* ── Toast notification ── */
  let _toastTimer;
  function showToast(msg, type = 'ok') {
    const existing = document.querySelector('.toast');
    if (existing) existing.remove();
    clearTimeout(_toastTimer);

    const t = document.createElement('div');
    t.className = 'toast ' + type;

    const icons = { ok: 'fa-circle-check', warn: 'fa-triangle-exclamation', err: 'fa-circle-xmark' };
    const cols  = { ok: 'color:var(--safe)', warn: 'color:var(--am2)', err: 'color:var(--danger)' };
    t.innerHTML = `<i class="fa-solid ${icons[type]}" style="${cols[type]}"></i>${msg}`;

    document.body.appendChild(t);
    _toastTimer = setTimeout(() => t.remove(), 3500);
  }

  /* ── Toolbar helpers ── */
  function doExport()   { showToast('Exporting current view as CSV…', 'ok');              }
  function runAIBatch() { showToast('AI batch review started — processing 7 documents…', 'ok'); }

  /* ── Tab switching ── */
  document.querySelectorAll('.tab').forEach(tab => {
    tab.addEventListener('click', function () {
      this.parentElement.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
      this.classList.add('active');
    });
  });
</script>
