export const ThemeContext = createContext();

export function ThemeProvider({ children }) {
  const [pastelMode, setPastelMode] = useState(true);
  const [dyslexiaFont, setDyslexiaFont] = useState(false);

  const theme = pastelMode
    ? {
        mode: "pastel",
        background: "#fff7fb",
        card: "#ffeef6",
        text: "#4b2a3a",
        accent: "#ffb3d9",
      }
    : {
        mode: "cozyDark",
        background: "#151515",
        card: "#262626",
        text: "#f2f2f2",
        accent: "#c084fc",
      };

  return (
    <ThemeContext.Provider
      value={{ theme, pastelMode, setPastelMode, dyslexiaFont, setDyslexiaFont }}
    >
      {children}
    </ThemeContext.Provider>
  );
}

  const { theme } = useContext(ThemeContext);

  // Simple paper theme augmentation

      text: theme.text,

  };

  return (

      <NavigationContainer>

            headerShown: false,
            tabBarStyle: {
              backgroundColor: theme.card,
              borderTopLeftRadius: 18,
              borderTopRightRadius: 18,
              height: 70,
              paddingBottom: 8,
            },
            tabBarActiveTintColor: theme.accent,
            tabBarInactiveTintColor: "#888",
            tabBarIcon: ({ color, size }) => {
              let iconName = "home";
              if (route.name === "Home") iconName = "home";
              if (route.name === "Blog") iconName = "book-open-page-variant";
              if (route.name === "Quizzes") iconName = "puzzle";
              if (route.name === "Chat") iconName = "chat";
              return <MaterialCommunityIcons name={iconName} size={size} color={color} />;
            },
          })}
        >
          <Tab.Screen name="Home" component={HomeScreen} />
          <Tab.Screen name="Blog" component={BlogScreen} />
          <Tab.Screen name="Quizzes" component={QuizScreen} />
          <Tab.Screen name="Chat" component={ChatScreen} />
        </Tab.Navigator>
      </NavigationContainer>
    </PaperProvider>
  );
}

import React, { useContext, useState } from "react";
import { View, Text, TouchableOpacity, StyleSheet } from "react-native";
import { ThemeContext } from "../components/ThemeContext";

const sampleQ = {
  q: "Which cozy space do you prefer?",
  options: ["Fairy Garden", "Tea Table", "Window Seat", "Comfy Couch"],
};

export default function QuizScreen() {
  const { theme } = useContext(ThemeContext);
  const [selected, setSelected] = useState(null);
  const [submitted, setSubmitted] = useState(false);

  const submit = () => {
    if (selected === null) return;
    setSubmitted(true);
  };

  return (
    <View style={[styles.container, { backgroundColor: theme.background, padding: 16 }]}>
      <Text style={[styles.q, { color: theme.text }]}>{sampleQ.q}</Text>

      {sampleQ.options.map((o, i) => {
        const active = selected === i;
        return (
          <TouchableOpacity
            key={o}
            onPress={() => setSelected(i)}
            style={[
              styles.option,
              { borderColor: active ? theme.accent : "#ddd", backgroundColor: active ? theme.card : "#fff" },
            ]}
          >
            <Text style={{ color: theme.text }}>{o}</Text>
          </TouchableOpacity>
        );
      })}

      <TouchableOpacity onPress={submit} style={[styles.submit, { backgroundColor: theme.accent }]}>
        <Text style={{ color: "#fff", fontWeight: "700" }}>{submitted ? "Submitted âœ“" : "Submit"}</Text>
      </TouchableOpacity>

      {submitted && selected !== null && (
        <View style={[styles.result, { borderColor: theme.accent }]}>
          <Text style={{ color: theme.text }}>Result: You picked "{sampleQ.options[selected]}" â€” cute choice! ðŸŒ¸</Text>
        </View>
      )}
    </View>
  );
}

import React, { useContext, useState, useRef } from "react";
import { View, TextInput, TouchableOpacity, Text, FlatList, KeyboardAvoidingView, Platform, StyleSheet } from "react-native";
import { ThemeContext } from "../components/ThemeContext";

export default function ChatScreen() {
  const { theme } = useContext(ThemeContext);
  const [messages, setMessages] = useState([
    { id: "1", text: "Welcome to the cottage chat! Be kind, be cozy ðŸ’–" },
  ]);
  const [text, setText] = useState("");

  const send = () => {
    if (!text.trim()) return;
    const newMsg = { id: Date.now().toString(), text };
    setMessages((m) => [newMsg, ...m]);
    setText("");
  };

  return (
    <KeyboardAvoidingView behavior={Platform.OS === "ios" ? "padding" : "height"} style={[styles.container, { backgroundColor: theme.background }]}>
      <FlatList
        data={messages}
        inverted
        keyExtractor={(i) => i.id}
        contentContainerStyle={{ padding: 12 }}
        renderItem={({ item }) => (
          <View style={[styles.msg, { backgroundColor: theme.card }]}>
            <Text style={{ color: theme.text }}>{item.text}</Text>
          </View>
        )}
      />
      <View style={styles.compose}>
        <TextInput
          placeholder="Say something cozy..."
          placeholderTextColor="#9a8a9a"
          value={text}
          onChangeText={setText}
          style={[styles.input, { color: theme.text }]}
        />
        <TouchableOpacity onPress={send} style={[styles.send, { backgroundColor: theme.accent }]}>
          <Text style={{ color: "#fff", fontWeight: "700" }}>Send</Text>
        </TouchableOpacity>
      </View>
    </KeyboardAvoidingView>
  );
}
