import React, { useState } from "react";
import { SafeAreaView, Text, FlatList, TouchableOpacity, View } from "react-native";

const antraege = [
  {
    id: "1",
    titel: "Bürgergeld beantragen",
    steps: [
      "Formular vom Jobcenter herunterladen oder online starten.",
      "Persönliche Daten eintragen (Name, Adresse, Geburtsdatum).",
      "Nachweise hochladen (Mietvertrag, Kontoauszüge, Arbeitsvertrag).",
      "Formular prüfen und abschicken.",
      "Termin beim Jobcenter wahrnehmen.",
    ],
  },
  {
    id: "2",
    titel: "Kindergeld beantragen",
    steps: [
      "Antragsformular der Familienkasse öffnen.",
      "Geburtsurkunde des Kindes bereithalten.",
      "Bankverbindung angeben.",
      "Antrag online oder per Post einreichen.",
      "Bewilligungsbescheid abwarten.",
    ],
  },
  {
    id: "3",
    titel: "BAföG beantragen",
    steps: [
      "Formblatt 1 ausfüllen (persönliche Angaben).",
      "Einkommensnachweise der Eltern hochladen.",
      "Mietvertrag / Studienbescheinigung beilegen.",
      "Antrag digital an das BAföG-Amt schicken.",
      "Status im BAföG-Portal prüfen.",
    ],
  },
];

export default function App() {
  const [auswahl, setAuswahl] = useState(null);

  return (
    <SafeAreaView style={{ flex: 1, padding: 20, backgroundColor: "#f4f4f4" }}>
      {!auswahl ? (
        <>
          <Text style={{ fontSize: 24, fontWeight: "bold", marginBottom: 20 }}>
            Wähle einen Antrag
          </Text>
          <FlatList
            data={antraege}
            keyExtractor={(item) => item.id}
            renderItem={({ item }) => (
              <TouchableOpacity
                style={{
                  backgroundColor: "white",
                  padding: 15,
                  marginBottom: 10,
                  borderRadius: 10,
                  shadowColor: "#000",
                  shadowOpacity: 0.1,
                  shadowRadius: 4,
                }}
                onPress={() => setAuswahl(item)}
              >
                <Text style={{ fontSize: 18 }}>{item.titel}</Text>
              </TouchableOpacity>
            )}
          />
        </>
      ) : (
        <View>
          <Text style={{ fontSize: 22, fontWeight: "bold", marginBottom: 10 }}>
            {auswahl.titel}
          </Text>
          {auswahl.steps.map((step, index) => (
            <Text key={index} style={{ fontSize: 16, marginVertical: 6 }}>
              {index + 1}. {step}
            </Text>
          ))}
          <TouchableOpacity
            style={{
              backgroundColor: "#007AFF",
              padding: 15,
              borderRadius: 10,
              marginTop: 20,
            }}
            onPress={() => setAuswahl(null)}
          >
            <Text style={{ color: "white", textAlign: "center", fontSize: 16 }}>
              Zurück zur Übersicht
            </Text>
          </TouchableOpacity>
        </View>
      )}
    </SafeAreaView>
  );
}
