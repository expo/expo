import React, { useState } from 'react';
import {
  Text,
  View,
  FlatList,
  TouchableOpacity,
  Image,
  TextInput,
  Linking,
  ScrollView
} from 'react-native';

export default function App() {

  const products = [
    {
      id: '1',
      name: 'Solar Inverter',
      price: '1299 SAR',
      image: 'https://via.placeholder.com/150'
    },
    {
      id: '2',
      name: 'LED Flood Light',
      price: '250 SAR',
      image: 'https://via.placeholder.com/150'
    },
    {
      id: '3',
      name: 'Power Bank',
      price: '99 SAR',
      image: 'https://via.placeholder.com/150'
    },
    {
      id: '4',
      name: 'Battery',
      price: '800 SAR',
      image: 'https://via.placeholder.com/150'
    }
  ];

  const [request, setRequest] = useState('');

  const orderOnWhatsApp = (product) => {
    const message = `Hello, I want to order: ${product.name} - ${product.price}`;
    Linking.openURL(`https://wa.me/923415725431?text=${encodeURIComponent(message)}`);
  };

  const sendRequest = () => {
    const message = `Customer Request: ${request}`;
    Linking.openURL(`https://wa.me/923415725431?text=${encodeURIComponent(message)}`);
  };

  return (
    <ScrollView style={{ padding: 15, marginTop: 40 }}>

      <Text style={{ fontSize: 26, fontWeight: 'bold', textAlign: 'center' }}>
        COFT PRIME
      </Text>

      <Text style={{ textAlign: 'center', marginBottom: 20 }}>
        Solar & Multi Products Store
      </Text>

      {/* PRODUCT LIST */}
      <FlatList
        data={products}
        keyExtractor={(item) => item.id}
        scrollEnabled={false}
        renderItem={({ item }) => (
          <View style={{
            backgroundColor: '#fff',
            padding: 10,
            marginBottom: 10,
            borderRadius: 10,
            elevation: 3
          }}>
            <Image
              source={{ uri: item.image }}
              style={{ height: 120, borderRadius: 10 }}
            />

            <Text style={{ fontSize: 18, marginTop: 5 }}>
              {item.name}
            </Text>

            <Text style={{ fontWeight: 'bold' }}>
              {item.price}
            </Text>

            <TouchableOpacity
              onPress={() => orderOnWhatsApp(item)}
              style={{
                backgroundColor: 'green',
                padding: 10,
                marginTop: 10,
                borderRadius: 5
              }}
            >
              <Text style={{ color: '#fff', textAlign: 'center' }}>
                Order on WhatsApp
              </Text>
            </TouchableOpacity>
          </View>
        )}
      />

      {/* REQUEST SECTION */}
      <Text style={{ marginTop: 20 }}>Request Any Product:</Text>

      <TextInput
        placeholder="Type your product..."
        value={request}
        onChangeText={setRequest}
        style={{
          borderWidth: 1,
          padding: 10,
          marginTop: 10,
          borderRadius: 5
        }}
      />

      <TouchableOpacity
        onPress={sendRequest}
        style={{
          backgroundColor: 'orange',
          padding: 12,
          marginTop: 10,
          borderRadius: 5
        }}
      >
        <Text style={{ textAlign: 'center', color: '#fff' }}>
          Send Request
        </Text>
      </TouchableOpacity>

    </ScrollView>
  );
}
