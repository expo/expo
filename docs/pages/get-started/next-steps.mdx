#### ৩. JavaScript (script.js)
```javascript
// Simple routing logic using hash-based navigation
window.addEventListener('hashchange', () => {
    loadContent();
});

function loadContent() {
    const contentDiv = document.querySelector('.content');
    const hash = window.location.hash;

    switch (hash) {
        case '#home':
            contentDiv.innerHTML = '<h1>Dashboard Overview</h1>';
            break;
        case '#add-money':
            contentDiv.innerHTML = '<h2>Add Money Request</h2>';
            break;
        case '#send-money':
            contentDiv.innerHTML = '<h2>Send Money Request</h2>';
            break;
        case '#cash-out':
            contentDiv.innerHTML = '<h2>Cash Out Request</h2>';
            break;
        case '#payment':
            contentDiv.innerHTML = '<h2>Payment Request</h2>';
            break;
        case '#bank-transfer':
            contentDiv.innerHTML = '<h2>Bank Transfer Request</h2>';
            break;
        case '#pay-bill':
            contentDiv.innerHTML = '<h2>Pay Bill Request</h2>';
            break;
        case '#loan':
            contentDiv.innerHTML = '<h2>Loan Request</h2>';
            break;
        case '#payment-confirmation':
            contentDiv.innerHTML = '<h2>Payment Confirmation</h2>';
            break;
        case '#transaction-history':
            contentDiv.innerHTML = '<h2>Transaction History</h2>';
            break;
        case '#notification':
            contentDiv.innerHTML = '<h2>Single Notification</h2>';
            break;
        case '#live-chat':
            contentDiv.innerHTML = '<h2>Live Chat</h2>';
            break;
        case '#helpline':
            contentDiv.innerHTML = '<h2>Helpline</h2>';
            break;
        case '#settings':
            contentDiv.innerHTML = '<h2>App Settings</h2>';
            break;
        case '#dollar-rate':
            contentDiv.innerHTML = '<h2>Dollar Rate</h2>';
            break;
        default:
            contentDiv.innerHTML = '<h1>Welcome to Admin Dashboard</h1>';
            break;
    }
}

// Load default content on page load
loadContent();
```
